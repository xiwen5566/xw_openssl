43 static int check_alerts(HANDSHAKE_RESULT *result, SSL_TEST_CTX *test_ctx)
44 {
45  if (result->client_alert_sent != result->client_alert_received) {
46  fprintf(stderr, "Client sent alert %s but server received %s\n.",
47  print_alert(result->client_alert_sent),
48  print_alert(result->client_alert_received));
49  /*
50  * We can't bail here because the peer doesn't always get far enough
51  * to process a received alert. Specifically, in protocol version
52  * negotiation tests, we have the following scenario.
53  * Client supports TLS v1.2 only; Server supports TLS v1.1.
54  * Client proposes TLS v1.2; server responds with 1.1;
55  * Client now sends a protocol alert, using TLS v1.2 in the header.
56  * The server, however, rejects the alert because of version mismatch
57  * in the record layer; therefore, the server appears to never
58  * receive the alert.
59  */
60  /* return 0; */
61  }
62 
63  if (result->server_alert_sent != result->server_alert_received) {
64  fprintf(stderr, "Server sent alert %s but client received %s\n.",
65  print_alert(result->server_alert_sent),
66  print_alert(result->server_alert_received));
67  /* return 0; */
68  }
69 
70  /* Tolerate an alert if one wasn't explicitly specified in the test. */
71  if (test_ctx->expected_client_alert
72  /*
73  * The info callback alert value is computed as
74  * (s->s3->send_alert[0] << 8) | s->s3->send_alert[1]
75  * where the low byte is the alert code and the high byte is other stuff.
76  */
77  && (result->client_alert_sent & 0xff) != test_ctx->expected_client_alert) {
78  fprintf(stderr, "ClientAlert mismatch: expected %s, got %s.\n",
79  print_alert(test_ctx->expected_client_alert),
80  print_alert(result->client_alert_sent));
81  return 0;
82  }
83 
84  if (test_ctx->expected_server_alert
85  && (result->server_alert_sent & 0xff) != test_ctx->expected_server_alert) {
86  fprintf(stderr, "ServerAlert mismatch: expected %s, got %s.\n",
87  print_alert(test_ctx->expected_server_alert),
88  print_alert(result->server_alert_sent));
89  return 0;
90  }
91 
92  if (result->client_num_fatal_alerts_sent > 1) {
93  fprintf(stderr, "Client sent %d fatal alerts.\n",
94  result->client_num_fatal_alerts_sent);
95  return 0;
96  }
97  if (result->server_num_fatal_alerts_sent > 1) {
98  fprintf(stderr, "Server sent %d alerts.\n",
99  result->server_num_fatal_alerts_sent);
100  return 0;
101  }
102  return 1;
103 }

248 /*
249  * This could be further simplified by constructing an expected
250  * HANDSHAKE_RESULT, and implementing comparison methods for
251  * its fields.
252  */
253 static int check_test(HANDSHAKE_RESULT *result, SSL_TEST_CTX *test_ctx)
254 {
255  int ret = 1;
256  ret &= check_result(result, test_ctx);
257  ret &= check_alerts(result, test_ctx);
258  if (result->result == SSL_TEST_SUCCESS) {
259  ret &= check_protocol(result, test_ctx);
260  ret &= check_servername(result, test_ctx);
261  ret &= check_session_ticket(result, test_ctx);
262  ret &= (result->session_ticket_do_not_call == 0);
263 #ifndef OPENSSL_NO_NEXTPROTONEG
264  ret &= check_npn(result, test_ctx);
265 #endif
266  ret &= check_alpn(result, test_ctx);
267  ret &= check_resumption(result, test_ctx);
268  ret &= check_tmp_key(result, test_ctx);
269  ret &= check_server_cert_type(result, test_ctx);
270  ret &= check_server_sign_hash(result, test_ctx);
271  ret &= check_server_sign_type(result, test_ctx);
272  ret &= check_client_cert_type(result, test_ctx);
273  ret &= check_client_sign_hash(result, test_ctx);
274  ret &= check_client_sign_type(result, test_ctx);
275  }
276  return ret;
277 }

279 static int test_handshake(int idx)
280 {
281  int ret = 0;
282  SSL_CTX *server_ctx = NULL, *server2_ctx = NULL, *client_ctx = NULL,
283  *resume_server_ctx = NULL, *resume_client_ctx = NULL;
284  SSL_TEST_CTX *test_ctx = NULL;
285  HANDSHAKE_RESULT *result = NULL;
286  char test_app[MAX_TESTCASE_NAME_LENGTH];
287 
288  BIO_snprintf(test_app, sizeof(test_app), "test-%d", idx);
289 
290  test_ctx = SSL_TEST_CTX_create(conf, test_app);
291  if (test_ctx == NULL)
292  goto err;
293 
294 #ifndef OPENSSL_NO_DTLS
295  if (test_ctx->method == SSL_TEST_METHOD_DTLS) {
296  server_ctx = SSL_CTX_new(DTLS_server_method());
297  if (test_ctx->extra.server.servername_callback !=
298  SSL_TEST_SERVERNAME_CB_NONE) {
299  server2_ctx = SSL_CTX_new(DTLS_server_method());
300  TEST_check(server2_ctx != NULL);
301  }
302  client_ctx = SSL_CTX_new(DTLS_client_method());
303  if (test_ctx->handshake_mode == SSL_TEST_HANDSHAKE_RESUME) {
304  resume_server_ctx = SSL_CTX_new(DTLS_server_method());
305  resume_client_ctx = SSL_CTX_new(DTLS_client_method());
306  TEST_check(resume_server_ctx != NULL);
307  TEST_check(resume_client_ctx != NULL);
308  }
309  }
310 #endif
311  if (test_ctx->method == SSL_TEST_METHOD_TLS) {
312  server_ctx = SSL_CTX_new(TLS_server_method());
313  /* SNI on resumption isn't supported/tested yet. */
314  if (test_ctx->extra.server.servername_callback !=
315  SSL_TEST_SERVERNAME_CB_NONE) {
316  server2_ctx = SSL_CTX_new(TLS_server_method());
317  TEST_check(server2_ctx != NULL);
318  }
319  client_ctx = SSL_CTX_new(TLS_client_method());
320 
321  if (test_ctx->handshake_mode == SSL_TEST_HANDSHAKE_RESUME) {
322  resume_server_ctx = SSL_CTX_new(TLS_server_method());
323  resume_client_ctx = SSL_CTX_new(TLS_client_method());
324  TEST_check(resume_server_ctx != NULL);
325  TEST_check(resume_client_ctx != NULL);
326  }
327  }
328 
329  TEST_check(server_ctx != NULL);
330  TEST_check(client_ctx != NULL);
331 
332  TEST_check(CONF_modules_load(conf, test_app, 0) > 0);
333 
334  if (!SSL_CTX_config(server_ctx, "server")
335  || !SSL_CTX_config(client_ctx, "client")) {
336  goto err;
337  }
338 
339  if (server2_ctx != NULL && !SSL_CTX_config(server2_ctx, "server2"))
340  goto err;
341  if (resume_server_ctx != NULL
342  && !SSL_CTX_config(resume_server_ctx, "resume-server"))
343  goto err;
344  if (resume_client_ctx != NULL
345  && !SSL_CTX_config(resume_client_ctx, "resume-client"))
346  goto err;
347 
348  result = do_handshake(server_ctx, server2_ctx, client_ctx,
349  resume_server_ctx, resume_client_ctx, test_ctx);
350 
351  ret = check_test(result, test_ctx);
352 
353 err:
354  CONF_modules_unload(0);
355  SSL_CTX_free(server_ctx);
356  SSL_CTX_free(server2_ctx);
357  SSL_CTX_free(client_ctx);
358  SSL_CTX_free(resume_server_ctx);
359  SSL_CTX_free(resume_client_ctx);
360  SSL_TEST_CTX_free(test_ctx);
361  HANDSHAKE_RESULT_free(result);
362  return ret;
363 }

365 int test_main(int argc, char **argv)
366 {
367  int result = 0;
368  long num_tests;
369 
370  if (argc != 2)
371  return 1;
372

375 
376  /* argv[1] should point to the test conf file */
377  TEST_check(NCONF_load(conf, argv[1], NULL) > 0);
378 
379  TEST_check(NCONF_get_number_e(conf, NULL, "num_tests", &num_tests));
380 
381  ADD_ALL_TESTS(test_handshake, (int)(num_tests));
382  result = run_tests(argv[0]);
383 
384  NCONF_free(conf);
385  return result;
386 }
