50

53

57  }

84 {

99  return (0);

107  for (i = a->top - 1; i >= 0; i--)

109  else

113  t[nw + i] = (l << lb) & BN_MASK2;

123 {

127

134  }

144  if (r != a) {

156  r->top = i;
157 
158  if (rb == 0) {
