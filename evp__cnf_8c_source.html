20 static int alg_module_init(CONF_IMODULE *md, const CONF *cnf)
21 {
22  int i;
23  const char *oid_section;
24  STACK_OF(CONF_VALUE) *sktmp;
25  CONF_VALUE *oval;
26 
27  oid_section = CONF_imodule_get_value(md);
28  if ((sktmp = NCONF_get_section(cnf, oid_section)) == NULL) {
29  EVPerr(EVP_F_ALG_MODULE_INIT, EVP_R_ERROR_LOADING_SECTION);
30  return 0;
31  }
32  for (i = 0; i < sk_CONF_VALUE_num(sktmp); i++) {
33  oval = sk_CONF_VALUE_value(sktmp, i);
34  if (strcmp(oval->name, "fips_mode") == 0) {
35  int m;
36  if (!X509V3_get_value_bool(oval, &m)) {
37  EVPerr(EVP_F_ALG_MODULE_INIT, EVP_R_INVALID_FIPS_MODE);
38  return 0;
39  }
40  if (m > 0) {
41 #ifdef OPENSSL_FIPS
42  if (!FIPS_mode() && !FIPS_mode_set(1)) {
43  EVPerr(EVP_F_ALG_MODULE_INIT,
44  EVP_R_ERROR_SETTING_FIPS_MODE);
45  return 0;
46  }
47 #else
48  EVPerr(EVP_F_ALG_MODULE_INIT, EVP_R_FIPS_MODE_NOT_SUPPORTED);
49  return 0;
50 #endif
51  }
52  } else {
53  EVPerr(EVP_F_ALG_MODULE_INIT, EVP_R_UNKNOWN_OPTION);
54  ERR_add_error_data(4, "name=", oval->name,
55  ", value=", oval->value);
56  }
57 
58  }
59  return 1;
60 }
