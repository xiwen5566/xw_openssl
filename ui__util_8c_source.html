147  || UI_method_set_reader(ui_method, ui_read) < 0
148  || UI_method_set_writer(ui_method, ui_write) < 0
149  || UI_method_set_closer(ui_method, ui_close) < 0
150  || !RUN_ONCE(&get_index_once, ui_method_data_index_init)
151  || UI_method_set_ex_data(ui_method, ui_method_data_index, data) < 0) {
152  UI_destroy_method(ui_method);
153  OPENSSL_free(data);
154  return NULL;
155  }
