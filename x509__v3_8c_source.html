162  ERR_R_MALLOC_FAILURE);

165  } else

168  if (!X509_EXTENSION_set_object(ret, obj))
169  goto err;

171  goto err;
172  if (!X509_EXTENSION_set_data(ret, data))
173  goto err;
174 
175  if ((ex != NULL) && (*ex == NULL))
176  *ex = ret;
177  return (ret);

180  X509_EXTENSION_free(ret);
