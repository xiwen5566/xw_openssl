26 static int pkey_poly1305_init(EVP_PKEY_CTX *ctx)

51 static int pkey_poly1305_copy(EVP_PKEY_CTX *dst, EVP_PKEY_CTX *src)
52 {
53  POLY1305_PKEY_CTX *sctx, *dctx;
54 
55  /* allocate memory for dst->data and a new POLY1305_CTX in dst->data->ctx */
56  if (!pkey_poly1305_init(dst))
57  return 0;
58  sctx = EVP_PKEY_CTX_get_data(src);
59  dctx = EVP_PKEY_CTX_get_data(dst);
60  if (ASN1_STRING_get0_data(&sctx->ktmp) != NULL &&
61  !ASN1_STRING_copy(&dctx->ktmp, &sctx->ktmp)) {
62  /* cleanup and free the POLY1305_PKEY_CTX in dst->data */
63  pkey_poly1305_cleanup(dst);
64  return 0;
65  }
66  memcpy(&dctx->ctx, &sctx->ctx, sizeof(POLY1305));
67  return 1;
68 }

70 static int pkey_poly1305_keygen(EVP_PKEY_CTX *ctx, EVP_PKEY *pkey)
71 {
72  ASN1_OCTET_STRING *key;
73  POLY1305_PKEY_CTX *pctx = EVP_PKEY_CTX_get_data(ctx);
74 
75  if (ASN1_STRING_get0_data(&pctx->ktmp) == NULL)
76  return 0;
77  key = ASN1_OCTET_STRING_dup(&pctx->ktmp);
78  if (key == NULL)
79  return 0;
80  return EVP_PKEY_assign_POLY1305(pkey, key);
81 }

91 static int poly1305_signctx_init(EVP_PKEY_CTX *ctx, EVP_MD_CTX *mctx)
92 {
93  POLY1305_PKEY_CTX *pctx = ctx->data;
94  ASN1_OCTET_STRING *key = (ASN1_OCTET_STRING *)ctx->pkey->pkey.ptr;
95 
96  if (key->length != POLY1305_KEY_SIZE)
97  return 0;
98  EVP_MD_CTX_set_flags(mctx, EVP_MD_CTX_FLAG_NO_INIT);
99  EVP_MD_CTX_set_update_fn(mctx, int_update);
100  Poly1305_Init(&pctx->ctx, key->data);
101  return 1;
102 }

114 static int pkey_poly1305_ctrl(EVP_PKEY_CTX *ctx, int type, int p1, void *p2)
115 {
116  POLY1305_PKEY_CTX *pctx = EVP_PKEY_CTX_get_data(ctx);
117  const unsigned char *key;
118  size_t len;
119 
120  switch (type) {
121 
122  case EVP_PKEY_CTRL_MD:
123  /* ignore */
124  break;
125 
126  case EVP_PKEY_CTRL_SET_MAC_KEY:
127  case EVP_PKEY_CTRL_DIGESTINIT:
128  if (type == EVP_PKEY_CTRL_SET_MAC_KEY) {
129  /* user explicitly setting the key */
130  key = p2;
131  len = p1;
132  } else {
133  /* user indirectly setting the key via EVP_DigestSignInit */
134  key = EVP_PKEY_get0_poly1305(EVP_PKEY_CTX_get0_pkey(ctx), &len);
135  }
136  if (key == NULL || len != POLY1305_KEY_SIZE ||
137  !ASN1_OCTET_STRING_set(&pctx->ktmp, key, len))
138  return 0;
139  Poly1305_Init(&pctx->ctx, ASN1_STRING_get0_data(&pctx->ktmp));
140  break;
141 
142  default:
143  return -2;
144 
145  }
146  return 1;
147 }
