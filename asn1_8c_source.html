187 static ASN1_PCTX *pctx;

189 int FuzzerInitialize(int *argc, char ***argv)
190 {
191  pctx = ASN1_PCTX_new();
192  ASN1_PCTX_set_flags(pctx, ASN1_PCTX_FLAGS_SHOW_ABSENT |
193  ASN1_PCTX_FLAGS_SHOW_SEQUENCE | ASN1_PCTX_FLAGS_SHOW_SSOF |
194  ASN1_PCTX_FLAGS_SHOW_TYPE | ASN1_PCTX_FLAGS_SHOW_FIELD_STRUCT_NAME);
195  ASN1_PCTX_set_str_flags(pctx, ASN1_STRFLGS_UTF8_CONVERT |
196  ASN1_STRFLGS_SHOW_TYPE | ASN1_STRFLGS_DUMP_ALL);
197 
198  OPENSSL_init_crypto(OPENSSL_INIT_LOAD_CRYPTO_STRINGS, NULL);
199  ERR_get_state();
200  CRYPTO_free_ex_index(0, -1);
201 
202  return 1;
203 }

205 int FuzzerTestOneInput(const uint8_t *buf, size_t len)
206 {
207  int n;
208 
209 
210  for (n = 0; item_type[n] != NULL; ++n) {
211  const uint8_t *b = buf;
212  unsigned char *der = NULL;
213  const ASN1_ITEM *i = ASN1_ITEM_ptr(item_type[n]);
214  ASN1_VALUE *o = ASN1_item_d2i(NULL, &b, len, i);
215 
216  if (o != NULL) {
217  BIO *bio = BIO_new(BIO_s_null());
218  ASN1_item_print(bio, o, 4, i, pctx);
219  BIO_free(bio);
220 
221  ASN1_item_i2d(o, &der, i);
222  OPENSSL_free(der);
223 
224  ASN1_item_free(o, i);
225  }
226  }
227 
228  ERR_clear_error();
229 
230  return 0;
231 }
