50  const unsigned char *iv, int enc)

52  if (!enc) {
53  if (EVP_CIPHER_CTX_mode(ctx) == EVP_CIPH_OFB_MODE)
54  enc = 1;
55  else if (EVP_CIPHER_CTX_mode(ctx) == EVP_CIPH_CFB_MODE)

62

65  OPENSSL_cleanse((unsigned char *)&tmp, sizeof(IDEA_KEY_SCHEDULE));
66  }
67  return 1;
