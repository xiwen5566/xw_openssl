52 DSA *get_dsa512()
53 {
54  DSA *dsa;
55  BIGNUM *priv_key, *pub_key, *p, *q, *g;
56 
57  if ((dsa = DSA_new()) == NULL)
58  return (NULL);
59  priv_key = BN_bin2bn(dsa512_priv, sizeof(dsa512_priv), NULL);
60  pub_key = BN_bin2bn(dsa512_pub, sizeof(dsa512_pub), NULL);
61  p = BN_bin2bn(dsa512_p, sizeof(dsa512_p), NULL);
62  q = BN_bin2bn(dsa512_q, sizeof(dsa512_q), NULL);
63  g = BN_bin2bn(dsa512_g, sizeof(dsa512_g), NULL);
64  if ((priv_key == NULL) || (pub_key == NULL) || (p == NULL) || (q == NULL)
65  || (g == NULL)) {
66  goto err;
67  }
68  if (!DSA_set0_pqg(dsa, p, q, g))
69  goto err;
70  p = q = g = NULL;
71 
72  if (!DSA_set0_key(dsa, pub_key, priv_key))
73  goto err;
74 
75  return dsa;
76  err:
77  DSA_free(dsa);
78  BN_free(priv_key);
79  BN_free(pub_key);
80  BN_free(p);
81  BN_free(q);
82  BN_free(g);
83  return NULL;
84 }
