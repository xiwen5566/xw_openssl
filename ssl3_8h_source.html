163 /* Maximum plaintext length: defined by SSL/TLS standards */
164 # define SSL3_RT_MAX_PLAIN_LENGTH 16384
165 /* Maximum compression overhead: defined by SSL/TLS standards */
166 # define SSL3_RT_MAX_COMPRESSED_OVERHEAD 1024
167 
168 /*
169  * The standards give a maximum encryption overhead of 1024 bytes. In
170  * practice the value is lower than this. The overhead is the maximum number
171  * of padding bytes (256) plus the mac size.
172  */
173 # define SSL3_RT_MAX_ENCRYPTED_OVERHEAD (256 + SSL3_RT_MAX_MD_SIZE)
174 
175 /*
176  * OpenSSL currently only uses a padding length of at most one block so the
177  * send overhead is smaller.
178  */
179 
180 # define SSL3_RT_SEND_MAX_ENCRYPTED_OVERHEAD \
181  (SSL_RT_MAX_CIPHER_BLOCK_SIZE + SSL3_RT_MAX_MD_SIZE)
182 
183 /* If compression isn't used don't include the compression overhead */
184 
185 # ifdef OPENSSL_NO_COMP
186 # define SSL3_RT_MAX_COMPRESSED_LENGTH SSL3_RT_MAX_PLAIN_LENGTH
187 # else
188 # define SSL3_RT_MAX_COMPRESSED_LENGTH \
189  (SSL3_RT_MAX_PLAIN_LENGTH+SSL3_RT_MAX_COMPRESSED_OVERHEAD)
190 # endif
191 # define SSL3_RT_MAX_ENCRYPTED_LENGTH \
192  (SSL3_RT_MAX_ENCRYPTED_OVERHEAD+SSL3_RT_MAX_COMPRESSED_LENGTH)
193 # define SSL3_RT_MAX_PACKET_SIZE \
