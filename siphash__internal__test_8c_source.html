41 static int benchmark_siphash(void)
42 {
43 # ifdef OPENSSL_CPUID_OBJ
44  SIPHASH siphash;
45  unsigned char key[SIPHASH_KEY_SIZE];
46  unsigned char buf[8192];
47  unsigned long long stopwatch;
48  unsigned long long OPENSSL_rdtsc();
49  unsigned int i;
50 
51  memset (buf,0x55,sizeof(buf));
52  memset (key,0xAA,sizeof(key));
53 
54  (void)SipHash_Init(&siphash, key, 0, 0, 0);
55 
56  for (i=0;i<100000;i++)
57  SipHash_Update(&siphash, buf, sizeof(buf));
58 
59  stopwatch = OPENSSL_rdtsc();
60  for (i=0;i<10000;i++)
61  SipHash_Update(&siphash, buf, sizeof(buf));
62  stopwatch = OPENSSL_rdtsc() - stopwatch;
63 
64  BIO_printf(b_stdout, "%g\n",stopwatch/(double)(i*sizeof(buf)));
65 
66  stopwatch = OPENSSL_rdtsc();
67  for (i=0;i<10000;i++) {
68  (void)SipHash_Init(&siphash, key, 0, 0, 0);
69  SipHash_Update(&siphash, buf, 16);
70  (void)SipHash_Final(&siphash, buf, SIPHASH_MAX_DIGEST_SIZE);
71  }
72  stopwatch = OPENSSL_rdtsc() - stopwatch;
73 
74  BIO_printf(b_stdout, "%g\n",stopwatch/(double)(i));
75 # else
76  BIO_printf(b_stderr,
77  "Benchmarking of siphash isn't available on this platform\n");
78 # endif
79  return 1;
80 }

236  unsigned char out[SIPHASH_MAX_DIGEST_SIZE];
237  size_t i;
238 
239  if (expectedlen != SIPHASH_MIN_DIGEST_SIZE &&
240  expectedlen != SIPHASH_MAX_DIGEST_SIZE)
241  return 0;
242 
243  if (inlen > sizeof(in))
244  return 0;
245 
246  /* key and in data are 00 01 02 ... */
247  for (i = 0; i < sizeof(key); i++)
248  key[i] = i;
249 
250  for (i = 0; i < inlen; i++)
251  in[i] = i;
252

254  BIO_printf(b_stderr, "SipHash_Init failed\n");
255  return 0;
256  }

258  if (!SipHash_Final(&siphash, out, expectedlen)) {
259  BIO_printf(b_stderr, "SipHash_Final failed\n");
260  return 0;
261  }
262

264  BIO_printf(b_stderr, "SipHash test #%d failed.\n", idx);
265  hex_out(b_stderr, "got: ", 16, out, expectedlen);
266  hex_out(b_stderr, "expected: ", 16, expected, expectedlen);

272  BIO_printf(b_stderr, "SipHash_Init failed\n");

292
