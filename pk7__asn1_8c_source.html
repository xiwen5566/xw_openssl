31 /* PKCS#7 streaming support */
32 static int pk7_cb(int operation, ASN1_VALUE **pval, const ASN1_ITEM *it,
33  void *exarg)
34 {
35  ASN1_STREAM_ARG *sarg = exarg;
36  PKCS7 **pp7 = (PKCS7 **)pval;
37 
38  switch (operation) {
39 
40  case ASN1_OP_STREAM_PRE:
41  if (PKCS7_stream(&sarg->boundary, *pp7) <= 0)
42  return 0;
43  case ASN1_OP_DETACHED_PRE:
44  sarg->ndef_bio = PKCS7_dataInit(*pp7, sarg->out);
45  if (!sarg->ndef_bio)
46  return 0;
47  break;
48 
49  case ASN1_OP_STREAM_POST:
50  case ASN1_OP_DETACHED_POST:
51  if (PKCS7_dataFinal(*pp7, sarg->ndef_bio) <= 0)
52  return 0;
53  break;
54 
55  }
56  return 1;
57 }
