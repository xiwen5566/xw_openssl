18 static int rsa_cb(int operation, ASN1_VALUE **pval, const ASN1_ITEM *it,
19  void *exarg)
20 {

24  return 2;
25  return 0;
26  } else if (operation == ASN1_OP_FREE_PRE) {

32 }
