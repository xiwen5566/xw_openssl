182 /*
183  * Glue an array of strings together. Return a BIO and put the string
184  * into |*out| so we can free it.
185  */
186 static BIO *glue(const char **pem, char **out)
187 {
188  char *dest;
189  int i;
190  size_t s = 0;
191 
192  /* Glue the strings together. */
193  for (i = 0; pem[i] != NULL; ++i)
194  s += strlen(pem[i]);
195  dest = *out = OPENSSL_malloc(s + 1);
196  if (dest == NULL)
197  return NULL;
198  for (i = 0; pem[i] != NULL; ++i)
199  dest += strlen(strcpy(dest, pem[i]));
200  return BIO_new_mem_buf(*out, s);
201 }

203 /*
204  * Create a CRL from an array of strings.
205  */
206 static X509_CRL *CRL_from_strings(const char **pem)
207 {
208  char *p;
209  BIO *b = glue(pem, &p);
210  X509_CRL *crl = PEM_read_bio_X509_CRL(b, NULL, NULL, NULL);
211 
212  OPENSSL_free(p);
213  BIO_free(b);
214  return crl;
215 }

231 /*
232  * Verify |leaf| certificate (chained up to |root|). |crls| if
233  * not NULL, is a list of CRLs to include in the verification. It is
234  * also free'd before returning, which is kinda yucky but convenient.
235  * Returns a value from X509_V_ERR_xxx or X509_V_OK.
236  */
237 static int verify(X509 *leaf, X509 *root, STACK_OF(X509_CRL) *crls,
238  unsigned long flags)
239 {
240  X509_STORE_CTX *ctx = X509_STORE_CTX_new();
241  X509_STORE *store = X509_STORE_new();
242  X509_VERIFY_PARAM *param = X509_VERIFY_PARAM_new();
243  STACK_OF(X509) *roots = sk_X509_new_null();
244  int status = X509_V_ERR_UNSPECIFIED;
245 
246  if (ctx == NULL || store == NULL || param == NULL || roots == NULL)
247  goto err;
248 
249  /* Create a stack; upref the cert because we free it below. */
250  X509_up_ref(root);
251  if (!sk_X509_push(roots, root))
252  goto err;
253 
254  if (!X509_STORE_CTX_init(ctx, store, leaf, NULL))
255  goto err;
256  X509_STORE_CTX_set0_trusted_stack(ctx, roots);
257  X509_STORE_CTX_set0_crls(ctx, crls);
258  X509_VERIFY_PARAM_set_time(param, PARAM_TIME);
259  if (X509_VERIFY_PARAM_get_time(param) != PARAM_TIME) {
260  fprintf(stderr, "set_time/get_time mismatch.\n");
261  goto err;
262  }
263  X509_VERIFY_PARAM_set_depth(param, 16);
264  if (flags)
265  X509_VERIFY_PARAM_set_flags(param, flags);
266  X509_STORE_CTX_set0_param(ctx, param);
267 
268  ERR_clear_error();
269  status = X509_verify_cert(ctx) == 1 ? X509_V_OK
270  : X509_STORE_CTX_get_error(ctx);
271 err:
272  sk_X509_pop_free(roots, X509_free);
273  sk_X509_CRL_pop_free(crls, X509_CRL_free);
274  X509_STORE_CTX_free(ctx);
275  X509_STORE_free(store);
276  return status;
277 }

279 /*
280  * Create a stack of CRL's. Upref each one because we call pop_free on
281  * the stack and need to keep the CRL's around until the test exits.
282  * Yes this crashes on malloc failure; it forces us to debug.
283  */
284 static STACK_OF(X509_CRL) *make_CRL_stack(X509_CRL *x1, X509_CRL *x2)
285 {
286  STACK_OF(X509_CRL) *sk = sk_X509_CRL_new_null();
287 
288  sk_X509_CRL_push(sk, x1);
289  X509_CRL_up_ref(x1);
290  if (x2 != NULL) {
291  sk_X509_CRL_push(sk, x2);
292  X509_CRL_up_ref(x2);
293  }
294  return sk;
295 }

297 static int test_crl()
298 {
299  X509 *root = X509_from_strings(kCRLTestRoot);
300  X509 *leaf = X509_from_strings(kCRLTestLeaf);
301  X509_CRL *basic_crl = CRL_from_strings(kBasicCRL);
302  X509_CRL *revoked_crl = CRL_from_strings(kRevokedCRL);
303  X509_CRL *bad_issuer_crl = CRL_from_strings(kBadIssuerCRL);
304  X509_CRL *known_critical_crl = CRL_from_strings(kKnownCriticalCRL);
305  X509_CRL *unknown_critical_crl = CRL_from_strings(kUnknownCriticalCRL);
306  X509_CRL *unknown_critical_crl2 = CRL_from_strings(kUnknownCriticalCRL2);
307  int status = 0;
308 
309  if (root == NULL || leaf == NULL || basic_crl == NULL
310  || revoked_crl == NULL || bad_issuer_crl == NULL
311  || known_critical_crl == NULL || unknown_critical_crl == NULL
312  || unknown_critical_crl2 == NULL) {
313  fprintf(stderr, "Failed to parse certificates and CRLs.\n");
314  goto err;
315  }
316 
317  if (verify(leaf, root, make_CRL_stack(basic_crl, NULL),
318  X509_V_FLAG_CRL_CHECK) != X509_V_OK) {
319  fprintf(stderr, "Cert with CRL didn't verify.\n");
320  goto err;
321  }
322 
323  if (verify(leaf, root, make_CRL_stack(basic_crl, revoked_crl),
324  X509_V_FLAG_CRL_CHECK) != X509_V_ERR_CERT_REVOKED) {
325  fprintf(stderr, "Revoked CRL wasn't checked.\n");
326  goto err;
327  }
