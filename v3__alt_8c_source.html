64 STACK_OF(CONF_VALUE) *i2v_GENERAL_NAME(X509V3_EXT_METHOD *method,
65  GENERAL_NAME *gen,
66  STACK_OF(CONF_VALUE) *ret)
67 {
68  unsigned char *p;
69  char oline[256], htmp[5];
70  int i;
71  switch (gen->type) {
72  case GEN_OTHERNAME:
73  X509V3_add_value("othername", "<unsupported>", &ret);
74  break;
75 
76  case GEN_X400:
77  X509V3_add_value("X400Name", "<unsupported>", &ret);
78  break;
79 
80  case GEN_EDIPARTY:
81  X509V3_add_value("EdiPartyName", "<unsupported>", &ret);
82  break;
83 
84  case GEN_EMAIL:
85  X509V3_add_value_uchar("email", gen->d.ia5->data, &ret);
86  break;
87 
88  case GEN_DNS:
89  X509V3_add_value_uchar("DNS", gen->d.ia5->data, &ret);
90  break;
91 
92  case GEN_URI:
93  X509V3_add_value_uchar("URI", gen->d.ia5->data, &ret);
94  break;
95 
96  case GEN_DIRNAME:
97  X509_NAME_oneline(gen->d.dirn, oline, 256);
98  X509V3_add_value("DirName", oline, &ret);
99  break;
100 
101  case GEN_IPADD:
102  p = gen->d.ip->data;
103  if (gen->d.ip->length == 4)
104  BIO_snprintf(oline, sizeof oline,
105  "%d.%d.%d.%d", p[0], p[1], p[2], p[3]);
106  else if (gen->d.ip->length == 16) {
107  oline[0] = 0;
108  for (i = 0; i < 8; i++) {
109  BIO_snprintf(htmp, sizeof htmp, "%X", p[0] << 8 | p[1]);
110  p += 2;
111  strcat(oline, htmp);
112  if (i != 7)
113  strcat(oline, ":");
114  }
115  } else {
116  X509V3_add_value("IP Address", "<invalid>", &ret);
117  break;
118  }
119  X509V3_add_value("IP Address", oline, &ret);
120  break;
121 
122  case GEN_RID:
123  i2t_ASN1_OBJECT(oline, 256, gen->d.rid);
124  X509V3_add_value("Registered ID", oline, &ret);
125  break;
126  }
127  return ret;
128 }

223 static int copy_issuer(X509V3_CTX *ctx, GENERAL_NAMES *gens)
224 {
225  GENERAL_NAMES *ialt;
226  GENERAL_NAME *gen;
227  X509_EXTENSION *ext;
228  int i;
229 
230  if (ctx && (ctx->flags == CTX_TEST))
231  return 1;
232  if (!ctx || !ctx->issuer_cert) {
233  X509V3err(X509V3_F_COPY_ISSUER, X509V3_R_NO_ISSUER_DETAILS);
234  goto err;
235  }
236  i = X509_get_ext_by_NID(ctx->issuer_cert, NID_subject_alt_name, -1);
237  if (i < 0)
238  return 1;
239  if ((ext = X509_get_ext(ctx->issuer_cert, i)) == NULL
240  || (ialt = X509V3_EXT_d2i(ext)) == NULL) {
241  X509V3err(X509V3_F_COPY_ISSUER, X509V3_R_ISSUER_DECODE_ERROR);
242  goto err;
243  }
244 
245  for (i = 0; i < sk_GENERAL_NAME_num(ialt); i++) {
246  gen = sk_GENERAL_NAME_value(ialt, i);
247  if (!sk_GENERAL_NAME_push(gens, gen)) {
248  X509V3err(X509V3_F_COPY_ISSUER, ERR_R_MALLOC_FAILURE);
249  goto err;
250  }
251  }
252  sk_GENERAL_NAME_free(ialt);
253 
254  return 1;
255 
256  err:
257  return 0;

300 static int copy_email(X509V3_CTX *ctx, GENERAL_NAMES *gens, int move_p)
301 {
302  X509_NAME *nm;
303  ASN1_IA5STRING *email = NULL;
304  X509_NAME_ENTRY *ne;
305  GENERAL_NAME *gen = NULL;
306  int i = -1;
307 
308  if (ctx != NULL && ctx->flags == CTX_TEST)
309  return 1;
310  if (ctx == NULL
311  || (ctx->subject_cert == NULL && ctx->subject_req == NULL)) {
312  X509V3err(X509V3_F_COPY_EMAIL, X509V3_R_NO_SUBJECT_DETAILS);
313  goto err;
314  }
315  /* Find the subject name */
316  if (ctx->subject_cert)
317  nm = X509_get_subject_name(ctx->subject_cert);
318  else
319  nm = X509_REQ_get_subject_name(ctx->subject_req);
320 
321  /* Now add any email address(es) to STACK */
322  while ((i = X509_NAME_get_index_by_NID(nm,
323  NID_pkcs9_emailAddress, i)) >= 0) {
324  ne = X509_NAME_get_entry(nm, i);
325  email = ASN1_STRING_dup(X509_NAME_ENTRY_get_data(ne));
326  if (move_p) {
327  X509_NAME_delete_entry(nm, i);
328  X509_NAME_ENTRY_free(ne);
329  i--;
330  }
331  if (email == NULL || (gen = GENERAL_NAME_new()) == NULL) {
332  X509V3err(X509V3_F_COPY_EMAIL, ERR_R_MALLOC_FAILURE);
333  goto err;
334  }
335  gen->d.ia5 = email;
336  email = NULL;
337  gen->type = GEN_EMAIL;
338  if (!sk_GENERAL_NAME_push(gens, gen)) {
339  X509V3err(X509V3_F_COPY_EMAIL, ERR_R_MALLOC_FAILURE);
340  goto err;
341  }
342  gen = NULL;
343  }
344 
345  return 1;
346 
347  err:
348  GENERAL_NAME_free(gen);
349  ASN1_IA5STRING_free(email);
350  return 0;
351 
352 }

384 GENERAL_NAME *a2i_GENERAL_NAME(GENERAL_NAME *out,
385  const X509V3_EXT_METHOD *method,
386  X509V3_CTX *ctx, int gen_type, const char *value,
387  int is_nc)
388 {
389  char is_string = 0;
390  GENERAL_NAME *gen = NULL;
391 
392  if (!value) {
393  X509V3err(X509V3_F_A2I_GENERAL_NAME, X509V3_R_MISSING_VALUE);
394  return NULL;
395  }
396 
397  if (out)
398  gen = out;
399  else {
400  gen = GENERAL_NAME_new();
401  if (gen == NULL) {
402  X509V3err(X509V3_F_A2I_GENERAL_NAME, ERR_R_MALLOC_FAILURE);
403  return NULL;
404  }
405  }
406 
407  switch (gen_type) {
408  case GEN_URI:
409  case GEN_EMAIL:
410  case GEN_DNS:
411  is_string = 1;
412  break;
413 
414  case GEN_RID:
415  {
416  ASN1_OBJECT *obj;
417  if ((obj = OBJ_txt2obj(value, 0)) == NULL) {
418  X509V3err(X509V3_F_A2I_GENERAL_NAME, X509V3_R_BAD_OBJECT);
419  ERR_add_error_data(2, "value=", value);
420  goto err;
421  }
422  gen->d.rid = obj;
423  }
424  break;
425 
426  case GEN_IPADD:
427  if (is_nc)
428  gen->d.ip = a2i_IPADDRESS_NC(value);
429  else
430  gen->d.ip = a2i_IPADDRESS(value);
431  if (gen->d.ip == NULL) {
432  X509V3err(X509V3_F_A2I_GENERAL_NAME, X509V3_R_BAD_IP_ADDRESS);
433  ERR_add_error_data(2, "value=", value);
434  goto err;
435  }
436  break;
437 
438  case GEN_DIRNAME:
439  if (!do_dirname(gen, value, ctx)) {
440  X509V3err(X509V3_F_A2I_GENERAL_NAME, X509V3_R_DIRNAME_ERROR);
441  goto err;
442  }
443  break;
444 
445  case GEN_OTHERNAME:
446  if (!do_othername(gen, value, ctx)) {
447  X509V3err(X509V3_F_A2I_GENERAL_NAME, X509V3_R_OTHERNAME_ERROR);
448  goto err;
449  }
450  break;
451  default:
452  X509V3err(X509V3_F_A2I_GENERAL_NAME, X509V3_R_UNSUPPORTED_TYPE);
453  goto err;
454  }
455 
456  if (is_string) {
457  if ((gen->d.ia5 = ASN1_IA5STRING_new()) == NULL ||
458  !ASN1_STRING_set(gen->d.ia5, (unsigned char *)value,
459  strlen(value))) {
460  X509V3err(X509V3_F_A2I_GENERAL_NAME, ERR_R_MALLOC_FAILURE);
461  goto err;
462  }
463  }
464 
465  gen->type = gen_type;
466 
467  return gen;
468 
469  err:
470  if (!out)
471  GENERAL_NAME_free(gen);
472  return NULL;
473 }

515 static int do_othername(GENERAL_NAME *gen, const char *value, X509V3_CTX *ctx)
516 {
517  char *objtmp = NULL, *p;
518  int objlen;
519 
520  if ((p = strchr(value, ';')) == NULL)
521  return 0;
522  if ((gen->d.otherName = OTHERNAME_new()) == NULL)
523  return 0;
524  /*
525  * Free this up because we will overwrite it. no need to free type_id
526  * because it is static
527  */
528  ASN1_TYPE_free(gen->d.otherName->value);
529  if ((gen->d.otherName->value = ASN1_generate_v3(p + 1, ctx)) == NULL)
530  return 0;
531  objlen = p - value;
532  objtmp = OPENSSL_strndup(value, objlen);
533  if (objtmp == NULL)
534  return 0;
535  gen->d.otherName->type_id = OBJ_txt2obj(objtmp, 0);
536  OPENSSL_free(objtmp);
537  if (!gen->d.otherName->type_id)
538  return 0;
539  return 1;
540 }
