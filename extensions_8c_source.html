47 /* Structure to define a built-in extension */
48 typedef struct extensions_definition_st {
49  /* The defined type for the extension */
50  unsigned int type;
51  /*
52  * The context that this extension applies to, e.g. what messages and
53  * protocol versions
54  */
55  unsigned int context;
56  /*
57  * Initialise extension before parsing. Always called for relevant contexts
58  * even if extension not present
59  */
60  int (*init)(SSL *s, unsigned int context);
61  /* Parse extension sent from client to server */
62  int (*parse_ctos)(SSL *s, PACKET *pkt, unsigned int context, X509 *x,
63  size_t chainidx, int *al);
64  /* Parse extension send from server to client */
65  int (*parse_stoc)(SSL *s, PACKET *pkt, unsigned int context, X509 *x,
66  size_t chainidx, int *al);
67  /* Construct extension sent from server to client */
68  int (*construct_stoc)(SSL *s, WPACKET *pkt, unsigned int context, X509 *x,
69  size_t chainidx, int *al);
70  /* Construct extension sent from client to server */
71  int (*construct_ctos)(SSL *s, WPACKET *pkt, unsigned int context, X509 *x,
72  size_t chainidx, int *al);
73  /*
74  * Finalise extension after parsing. Always called where an extensions was
75  * initialised even if the extension was not present. |sent| is set to 1 if
76  * the extension was seen, or 0 otherwise.
77  */
78  int (*final)(SSL *s, unsigned int context, int sent, int *al);
79 } EXTENSION_DEFINITION;

82  * Definitions of all built-in extensions. NOTE: Changes in the number or order
83  * of these extensions should be mirrored with equivalent changes to the
84  * indexes ( TLSEXT_IDX_* ) defined in ssl_locl.h.
85  * Each extension has an initialiser, a client and
86  * server side parser and a finaliser. The initialiser is called (if the
87  * extension is relevant to the given context) even if we did not see the
88  * extension in the message that we received. The parser functions are only
89  * called if we see the extension in the message. The finalisers are always
90  * called if the initialiser was called.

292 /*
293  * Verify whether we are allowed to use the extension |type| in the current
294  * |context|. Returns 1 to indicate the extension is allowed or unknown or 0 to
295  * indicate the extension is not allowed. If returning 1 then |*found| is set to
296  * 1 if we found a definition for the extension, and |*idx| is set to its index
297  */
298 static int verify_extension(SSL *s, unsigned int context, unsigned int type,
299  custom_ext_methods *meths, RAW_EXTENSION *rawexlist,
300  RAW_EXTENSION **found)
301 {
302  size_t i;
303  size_t builtin_num = OSSL_NELEM(ext_defs);
304  const EXTENSION_DEFINITION *thisext;
305 
306  for (i = 0, thisext = ext_defs; i < builtin_num; i++, thisext++) {
307  if (type == thisext->type) {
308  /* Check we're allowed to use this extension in this context */
309  if ((context & thisext->context) == 0)
310  return 0;
311 
312  if (SSL_IS_DTLS(s)) {
313  if ((thisext->context & EXT_TLS_ONLY) != 0)
314  return 0;
315  } else if ((thisext->context & EXT_DTLS_ONLY) != 0) {
316  return 0;
317  }
318 
319  *found = &rawexlist[i];
320  return 1;
321  }
322  }
323 
324  if ((context & (EXT_CLIENT_HELLO | EXT_TLS1_2_SERVER_HELLO)) == 0) {
325  /*
326  * Custom extensions only apply to <=TLS1.2. This extension is unknown
327  * in this context - we allow it
328  */
329  *found = NULL;
330  return 1;
331  }
332 
333  /* Check the custom extensions */
334  if (meths != NULL) {
335  for (i = builtin_num; i < builtin_num + meths->meths_count; i++) {
336  if (meths->meths[i - builtin_num].ext_type == type) {
337  *found = &rawexlist[i];
338  return 1;
339  }
340  }
341  }
342 
343  /* Unknown extension. We allow it */
344  *found = NULL;
345  return 1;
346 }

348 /*
349  * Check whether the context defined for an extension |extctx| means whether
350  * the extension is relevant for the current context |thisctx| or not. Returns
351  * 1 if the extension is relevant for this context, and 0 otherwise
352  */
353 static int extension_is_relevant(SSL *s, unsigned int extctx,
354  unsigned int thisctx)
355 {
356  if ((SSL_IS_DTLS(s)
357  && (extctx & EXT_TLS_IMPLEMENTATION_ONLY) != 0)
358  || (s->version == SSL3_VERSION
359  && (extctx & EXT_SSL3_ALLOWED) == 0)
360  || (SSL_IS_TLS13(s)
361  && (extctx & EXT_TLS1_2_AND_BELOW_ONLY) != 0)
362  || (!SSL_IS_TLS13(s) && (extctx & EXT_TLS1_3_ONLY) != 0))
363  return 0;
364 
365  return 1;
366 }

368 /*
369  * Gather a list of all the extensions from the data in |packet]. |context|
370  * tells us which message this extension is for. The raw extension data is
371  * stored in |*res| on success. In the event of an error the alert type to use
372  * is stored in |*al|. We don't actually process the content of the extensions
373  * yet, except to check their types. This function also runs the initialiser
374  * functions for all known extensions (whether we have collected them or not).
375  * If successful the caller is responsible for freeing the contents of |*res|.
376  *
377  * Per http://tools.ietf.org/html/rfc5246#section-7.4.1.4, there may not be
378  * more than one extension of the same type in a ClientHello or ServerHello.
379  * This function returns 1 if all extensions are unique and we have parsed their
380  * types, and 0 if the extensions contain duplicates, could not be successfully
381  * found, or an internal error occurred. We only check duplicates for
382  * extensions that we know about. We ignore others.
383  */
384 int tls_collect_extensions(SSL *s, PACKET *packet, unsigned int context,
385  RAW_EXTENSION **res, int *al)
386 {
387  PACKET extensions = *packet;
388  size_t i = 0;
389  custom_ext_methods *exts = NULL;
390  RAW_EXTENSION *raw_extensions = NULL;
391  const EXTENSION_DEFINITION *thisexd;
392 
393  *res = NULL;
394 
395  /*
396  * Initialise server side custom extensions. Client side is done during
397  * construction of extensions for the ClientHello.
398  */
399  if ((context & EXT_CLIENT_HELLO) != 0) {
400  exts = &s->cert->srv_ext;
401  custom_ext_init(&s->cert->srv_ext);
402  } else if ((context & EXT_TLS1_2_SERVER_HELLO) != 0) {
403  exts = &s->cert->cli_ext;
404  }
405 
406  raw_extensions = OPENSSL_zalloc((OSSL_NELEM(ext_defs)
407  + (exts != NULL ? exts->meths_count : 0))
408  * sizeof(*raw_extensions));
409  if (raw_extensions == NULL) {
410  *al = SSL_AD_INTERNAL_ERROR;
411  SSLerr(SSL_F_TLS_COLLECT_EXTENSIONS, ERR_R_MALLOC_FAILURE);
412  return 0;
413  }
414 
415  while (PACKET_remaining(&extensions) > 0) {
416  unsigned int type;
417  PACKET extension;
418  RAW_EXTENSION *thisex;
419 
420  if (!PACKET_get_net_2(&extensions, &type) ||
421  !PACKET_get_length_prefixed_2(&extensions, &extension)) {
422  SSLerr(SSL_F_TLS_COLLECT_EXTENSIONS, SSL_R_BAD_EXTENSION);
423  *al = SSL_AD_DECODE_ERROR;
424  goto err;
425  }
426  /*
427  * Verify this extension is allowed. We only check duplicates for
428  * extensions that we recognise.
429  */
430  if (!verify_extension(s, context, type, exts, raw_extensions, &thisex)
431  || (thisex != NULL && thisex->present == 1)) {
432  SSLerr(SSL_F_TLS_COLLECT_EXTENSIONS, SSL_R_BAD_EXTENSION);
433  *al = SSL_AD_ILLEGAL_PARAMETER;
434  goto err;
435  }
436  if (thisex != NULL) {
437  thisex->data = extension;
438  thisex->present = 1;
439  thisex->type = type;
440  }
441  }
442 
443  /*
444  * Initialise all known extensions relevant to this context, whether we have
445  * found them or not
446  */
447  for (thisexd = ext_defs, i = 0; i < OSSL_NELEM(ext_defs); i++, thisexd++) {
448  if(thisexd->init != NULL && (thisexd->context & context) != 0
449  && extension_is_relevant(s, thisexd->context, context)
450  && !thisexd->init(s, context)) {
451  *al = SSL_AD_INTERNAL_ERROR;
452  goto err;
453  }
454  }
455 
456  *res = raw_extensions;
457  return 1;
458 
459  err:
460  OPENSSL_free(raw_extensions);
461  return 0;
462 }

464 /*
465  * Runs the parser for a given extension with index |idx|. |exts| contains the
466  * list of all parsed extensions previously collected by
467  * tls_collect_extensions(). The parser is only run if it is applicable for the
468  * given |context| and the parser has not already been run. If this is for a
469  * Certificate message, then we also provide the parser with the relevant
470  * Certificate |x| and its position in the |chainidx| with 0 being the first
471  * Certificate. Returns 1 on success or 0 on failure. In the event of a failure
472  * |*al| is populated with a suitable alert code. If an extension is not present
473  * this counted as success.
474  */
475 int tls_parse_extension(SSL *s, TLSEXT_INDEX idx, int context,
476  RAW_EXTENSION *exts, X509 *x, size_t chainidx, int *al)
477 {
478  RAW_EXTENSION *currext = &exts[idx];
479  int (*parser)(SSL *s, PACKET *pkt, unsigned int context, X509 *x,
480  size_t chainidx, int *al) = NULL;
481 
482  /* Skip if the extension is not present */
483  if (!currext->present)
484  return 1;
485 
486  if (s->ext.debug_cb)
487  s->ext.debug_cb(s, !s->server, currext->type,
488  PACKET_data(&currext->data),
489  PACKET_remaining(&currext->data),
490  s->ext.debug_arg);
491 
492  /* Skip if we've already parsed this extension */
493  if (currext->parsed)
494  return 1;
495 
496  currext->parsed = 1;
497 
498  if (idx < OSSL_NELEM(ext_defs)) {
499  /* We are handling a built-in extension */
500  const EXTENSION_DEFINITION *extdef = &ext_defs[idx];
501 
502  /* Check if extension is defined for our protocol. If not, skip */
503  if (!extension_is_relevant(s, extdef->context, context))
504  return 1;
505 
506  parser = s->server ? extdef->parse_ctos : extdef->parse_stoc;
507 
508  if (parser != NULL)
509  return parser(s, &currext->data, context, x, chainidx, al);
510 
511  /*
512  * If the parser is NULL we fall through to the custom extension
513  * processing
514  */
515  }
516 
517  /*
518  * This is a custom extension. We only allow this if it is a non
519  * resumed session on the server side.
520  *chain
521  * TODO(TLS1.3): We only allow old style <=TLS1.2 custom extensions.
522  * We're going to need a new mechanism for TLS1.3 to specify which
523  * messages to add the custom extensions to.
524  */
525  if ((!s->hit || !s->server)
526  && (context
527  & (EXT_CLIENT_HELLO | EXT_TLS1_2_SERVER_HELLO)) != 0
528  && custom_ext_parse(s, s->server, currext->type,
529  PACKET_data(&currext->data),
530  PACKET_remaining(&currext->data),
531  al) <= 0)
532  return 0;
533 
534  return 1;
535 }

537 /*
538  * Parse all remaining extensions that have not yet been parsed. Also calls the
539  * finalisation for all extensions at the end, whether we collected them or not.
540  * Returns 1 for success or 0 for failure. If we are working on a Certificate
541  * message then we also pass the Certificate |x| and its position in the
542  * |chainidx|, with 0 being the first certificate. On failure, |*al| is
543  * populated with a suitable alert code.
544  */
545 int tls_parse_all_extensions(SSL *s, int context, RAW_EXTENSION *exts, X509 *x,
546  size_t chainidx, int *al)
547 {
548  size_t i, numexts = OSSL_NELEM(ext_defs);
549  const EXTENSION_DEFINITION *thisexd;
550 
551  /* Calculate the number of extensions in the extensions list */
552  if ((context & EXT_CLIENT_HELLO) != 0) {
553  numexts += s->cert->srv_ext.meths_count;
554  } else if ((context & EXT_TLS1_2_SERVER_HELLO) != 0) {
555  numexts += s->cert->cli_ext.meths_count;
556  }
557 
558  /* Parse each extension in turn */
559  for (i = 0; i < numexts; i++) {
560  if (!tls_parse_extension(s, i, context, exts, x, chainidx, al))

580  * them to |pkt|. If this is an extension for a Certificate in a Certificate

582  * |chainidx| will indicate the position in the chainidx we are processing (with

587 int tls_construct_extensions(SSL *s, WPACKET *pkt, unsigned int context,

601  /*

603  * extensions length bytes to a ClientHello/ServerHello in SSLv3
604  */
605  || ((context & (EXT_CLIENT_HELLO | EXT_TLS1_2_SERVER_HELLO)) != 0
606  && s->version == SSL3_VERSION

608  WPACKET_FLAGS_ABANDON_ON_ZERO_LENGTH))) {

610  goto err;
611  }

614  reason = ssl_get_client_min_max_version(s, &min_version, &max_version);
615  if (reason != 0) {

628  * parsing.

631  * for TLS1.3, so that custom extensions can specify which of the
632  * multiple message they wish to add themselves to.
633  */
634  addcustom = 1;
635  }

638  SSLerr(SSL_F_TLS_CONSTRUCT_EXTENSIONS, ERR_R_INTERNAL_ERROR);

656  != 0)

665  || ((thisexd->context & EXT_TLS1_3_ONLY) != 0
666  && (context & EXT_CLIENT_HELLO) != 0
667  && (SSL_IS_DTLS(s) || max_version < TLS1_3_VERSION))

1093 int tls_psk_do_binder(SSL *s, const EVP_MD *md, const unsigned char *msgstart,
1094  size_t binderoffset, const unsigned char *binderin,
1095  unsigned char *binderout,
1096  SSL_SESSION *sess, int sign)
1097 {
1098  EVP_PKEY *mackey = NULL;
1099  EVP_MD_CTX *mctx = NULL;
1100  unsigned char hash[EVP_MAX_MD_SIZE], binderkey[EVP_MAX_MD_SIZE];
1101  unsigned char finishedkey[EVP_MAX_MD_SIZE], tmpbinder[EVP_MAX_MD_SIZE];
1102  const char resumption_label[] = "resumption psk binder key";
1103  size_t bindersize, hashsize = EVP_MD_size(md);
1104  int ret = -1;
1105 
1106  /* Generate the early_secret */
1107  if (!tls13_generate_secret(s, md, NULL, sess->master_key,
1108  sess->master_key_length,
1109  (unsigned char *)&s->early_secret)) {
1110  SSLerr(SSL_F_TLS_PSK_DO_BINDER, ERR_R_INTERNAL_ERROR);
1111  goto err;
1112  }
1113 
1114  /*
1115  * Create the handshake hash for the binder key...the messages so far are
1116  * empty!
1117  */
1118  mctx = EVP_MD_CTX_new();
1119  if (mctx == NULL
1120  || EVP_DigestInit_ex(mctx, md, NULL) <= 0
1121  || EVP_DigestFinal_ex(mctx, hash, NULL) <= 0) {
1122  SSLerr(SSL_F_TLS_PSK_DO_BINDER, ERR_R_INTERNAL_ERROR);
1123  goto err;
1124  }
1125 
1126  /* Generate the binder key */
1127  if (!tls13_hkdf_expand(s, md, s->early_secret,
1128  (unsigned char *)resumption_label,
1129  sizeof(resumption_label) - 1, hash, binderkey,
1130  hashsize)) {
1131  SSLerr(SSL_F_TLS_PSK_DO_BINDER, ERR_R_INTERNAL_ERROR);
1132  goto err;
1133  }
1134 
1135  /* Generate the finished key */
1136  if (!tls13_derive_finishedkey(s, md, binderkey, finishedkey, hashsize)) {
1137  SSLerr(SSL_F_TLS_PSK_DO_BINDER, ERR_R_INTERNAL_ERROR);
1138  goto err;
1139  }
1140 
1141  if (EVP_DigestInit_ex(mctx, md, NULL) <= 0) {
1142  SSLerr(SSL_F_TLS_PSK_DO_BINDER, ERR_R_INTERNAL_ERROR);
1143  goto err;
1144  }
1145 
1146  /*
1147  * Get a hash of the ClientHello up to the start of the binders. If we are
1148  * following a HelloRetryRequest then this includes the hash of the first
1149  * ClientHello and the HelloRetryRequest itself.
1150  */
1151  if (s->hello_retry_request) {
1152  size_t hdatalen;
1153  void *hdata;
1154 
1155  hdatalen = BIO_get_mem_data(s->s3->handshake_buffer, &hdata);
1156  if (hdatalen <= 0) {
1157  SSLerr(SSL_F_TLS_PSK_DO_BINDER, SSL_R_BAD_HANDSHAKE_LENGTH);
1158  goto err;
1159  }
1160 
1161  /*
1162  * For servers the handshake buffer data will include the second
1163  * ClientHello - which we don't want - so we need to take that bit off.
1164  */
1165  if (s->server) {
1166  if (hdatalen < s->init_num + SSL3_HM_HEADER_LENGTH) {
1167  SSLerr(SSL_F_TLS_PSK_DO_BINDER, ERR_R_INTERNAL_ERROR);
1168  goto err;
1169  }
1170  hdatalen -= s->init_num + SSL3_HM_HEADER_LENGTH;
1171  }
1172 
1173  if (EVP_DigestUpdate(mctx, hdata, hdatalen) <= 0) {
1174  SSLerr(SSL_F_TLS_PSK_DO_BINDER, ERR_R_INTERNAL_ERROR);
1175  goto err;
1176  }
1177  }
1178 
1179  if (EVP_DigestUpdate(mctx, msgstart, binderoffset) <= 0
1180  || EVP_DigestFinal_ex(mctx, hash, NULL) <= 0) {
1181  SSLerr(SSL_F_TLS_PSK_DO_BINDER, ERR_R_INTERNAL_ERROR);
1182  goto err;
1183  }
1184 
1185  mackey = EVP_PKEY_new_mac_key(EVP_PKEY_HMAC, NULL, finishedkey, hashsize);
1186  if (mackey == NULL) {
1187  SSLerr(SSL_F_TLS_PSK_DO_BINDER, ERR_R_INTERNAL_ERROR);
1188  goto err;
1189  }
1190 
1191  if (!sign)
1192  binderout = tmpbinder;
1193
