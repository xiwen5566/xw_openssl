70 int ec_main(int argc, char **argv)
71 {
72  BIO *in = NULL, *out = NULL;
73  ENGINE *e = NULL;
74  EC_KEY *eckey = NULL;
75  const EC_GROUP *group;
76  const EVP_CIPHER *enc = NULL;
77  point_conversion_form_t form = POINT_CONVERSION_UNCOMPRESSED;
78  char *infile = NULL, *outfile = NULL, *prog;
79  char *passin = NULL, *passout = NULL, *passinarg = NULL, *passoutarg = NULL;
80  OPTION_CHOICE o;
81  int asn1_flag = OPENSSL_EC_NAMED_CURVE, new_form = 0, new_asn1_flag = 0;
82  int informat = FORMAT_PEM, outformat = FORMAT_PEM, text = 0, noout = 0;
83  int pubin = 0, pubout = 0, param_out = 0, i, ret = 1, private = 0;
84  int no_public = 0, check = 0;
85 
86  prog = opt_init(argc, argv, ec_options);
87  while ((o = opt_next()) != OPT_EOF) {
88  switch (o) {
89  case OPT_EOF:
90  case OPT_ERR:
91  opthelp:
92  BIO_printf(bio_err, "%s: Use -help for summary.\n", prog);
93  goto end;
94  case OPT_HELP:
95  opt_help(ec_options);
96  ret = 0;
97  goto end;
98  case OPT_INFORM:
99  if (!opt_format(opt_arg(), OPT_FMT_ANY, &informat))
100  goto opthelp;
101  break;
102  case OPT_IN:
103  infile = opt_arg();
104  break;
105  case OPT_OUTFORM:
106  if (!opt_format(opt_arg(), OPT_FMT_PEMDER, &outformat))
107  goto opthelp;
108  break;
109  case OPT_OUT:
110  outfile = opt_arg();
111  break;
112  case OPT_NOOUT:
113  noout = 1;
114  break;
115  case OPT_TEXT:
116  text = 1;
117  break;
118  case OPT_PARAM_OUT:
119  param_out = 1;
120  break;
121  case OPT_PUBIN:
122  pubin = 1;
123  break;
124  case OPT_PUBOUT:
125  pubout = 1;
126  break;
127  case OPT_PASSIN:
128  passinarg = opt_arg();
129  break;
130  case OPT_PASSOUT:
131  passoutarg = opt_arg();
132  break;
133  case OPT_ENGINE:
134  e = setup_engine(opt_arg(), 0);
135  break;
136  case OPT_CIPHER:
137  if (!opt_cipher(opt_unknown(), &enc))
138  goto opthelp;
139  break;
140  case OPT_CONV_FORM:
141  if (!opt_pair(opt_arg(), conv_forms, &i))
142  goto opthelp;
143  new_form = 1;
144  form = i;
145  break;
146  case OPT_PARAM_ENC:
147  if (!opt_pair(opt_arg(), param_enc, &i))
148  goto opthelp;
149  new_asn1_flag = 1;
150  asn1_flag = i;
151  break;
152  case OPT_NO_PUBLIC:
153  no_public = 1;
154  break;
155  case OPT_CHECK:
156  check = 1;
157  break;
158  }
159  }
160  argc = opt_num_rest();
161  if (argc != 0)
162  goto opthelp;
163 
164  private = param_out || pubin || pubout ? 0 : 1;
165  if (text && !pubin)
166  private = 1;
167 
168  if (!app_passwd(passinarg, passoutarg, &passin, &passout)) {
169  BIO_printf(bio_err, "Error getting passwords\n");
170  goto end;
171  }
172 
173  if (informat != FORMAT_ENGINE) {
174  in = bio_open_default(infile, 'r', informat);
175  if (in == NULL)
176  goto end;
177  }
178 
179  BIO_printf(bio_err, "read EC key\n");
180  if (informat == FORMAT_ASN1) {
181  if (pubin)
182  eckey = d2i_EC_PUBKEY_bio(in, NULL);
183  else
184  eckey = d2i_ECPrivateKey_bio(in, NULL);
185  } else if (informat == FORMAT_ENGINE) {
186  EVP_PKEY *pkey;
187  if (pubin)
188  pkey = load_pubkey(infile, informat, 1, passin, e, "Public Key");
189  else
190  pkey = load_key(infile, informat, 1, passin, e, "Private Key");
191  if (pkey != NULL) {
192  eckey = EVP_PKEY_get1_EC_KEY(pkey);
193  EVP_PKEY_free(pkey);
194  }
195  } else {
196  if (pubin)
197  eckey = PEM_read_bio_EC_PUBKEY(in, NULL, NULL, NULL);
198  else
199  eckey = PEM_read_bio_ECPrivateKey(in, NULL, NULL, passin);
200  }
201  if (eckey == NULL) {
202  BIO_printf(bio_err, "unable to load Key\n");
203  ERR_print_errors(bio_err);
204  goto end;
205  }
206 
207  out = bio_open_owner(outfile, outformat, private);
208  if (out == NULL)
209  goto end;
210 
211  group = EC_KEY_get0_group(eckey);
212 
213  if (new_form)
214  EC_KEY_set_conv_form(eckey, form);
215 
216  if (new_asn1_flag)
217  EC_KEY_set_asn1_flag(eckey, asn1_flag);
218 
219  if (no_public)
220  EC_KEY_set_enc_flags(eckey, EC_PKEY_NO_PUBKEY);
221 
222  if (text) {
223  assert(pubin || private);
224  if (!EC_KEY_print(out, eckey, 0)) {
225  perror(outfile);
226  ERR_print_errors(bio_err);
227  goto end;
228  }
229  }
230 
231  if (check) {
232  if (EC_KEY_check_key(eckey) == 1) {
233  BIO_printf(bio_err, "EC Key valid.\n");
234  } else {
235  BIO_printf(bio_err, "EC Key Invalid!\n");
236  ERR_print_errors(bio_err);
237  }
238  }
239 
240  if (noout) {
241  ret = 0;
242  goto end;
243  }
244 
245  BIO_printf(bio_err, "writing EC key\n");
246  if (outformat == FORMAT_ASN1) {
247  if (param_out)
248  i = i2d_ECPKParameters_bio(out, group);
249  else if (pubin || pubout)
250  i = i2d_EC_PUBKEY_bio(out, eckey);
251  else {
252  assert(private);
253  i = i2d_ECPrivateKey_bio(out, eckey);
254  }
255  } else {
256  if (param_out)
257  i = PEM_write_bio_ECPKParameters(out, group);
258  else if (pubin || pubout)
259  i = PEM_write_bio_EC_PUBKEY(out, eckey);
260  else {
261  assert(private);
262  i = PEM_write_bio_ECPrivateKey(out, eckey, enc,
263  NULL, 0, NULL, passout);
264  }
265  }
266 
267  if (!i) {
268  BIO_printf(bio_err, "unable to write private key\n");
269  ERR_print_errors(bio_err);
270  } else
271  ret = 0;
272  end:
273  BIO_free(in);
274  BIO_free_all(out);
275  EC_KEY_free(eckey);
276  release_engine(e);
277  OPENSSL_free(passin);
278  OPENSSL_free(passout);
279  return (ret);
280 }
