197 /* Free up RecipientInfo additional data */
198 static int cms_ri_cb(int operation, ASN1_VALUE **pval, const ASN1_ITEM *it,
199  void *exarg)
200 {
201  if (operation == ASN1_OP_FREE_PRE) {
202  CMS_RecipientInfo *ri = (CMS_RecipientInfo *)*pval;
203  if (ri->type == CMS_RECIPINFO_TRANS) {
204  CMS_KeyTransRecipientInfo *ktri = ri->d.ktri;
205  EVP_PKEY_free(ktri->pkey);
206  X509_free(ktri->recip);
207  EVP_PKEY_CTX_free(ktri->pctx);
208  } else if (ri->type == CMS_RECIPINFO_KEK) {
209  CMS_KEKRecipientInfo *kekri = ri->d.kekri;
210  OPENSSL_clear_free(kekri->key, kekri->keylen);
211  } else if (ri->type == CMS_RECIPINFO_PASS) {
212  CMS_PasswordRecipientInfo *pwri = ri->d.pwri;
213  OPENSSL_clear_free(pwri->pass, pwri->passlen);
214  }
215  }
216  return 1;
217 }

280 /* CMS streaming support */
281 static int cms_cb(int operation, ASN1_VALUE **pval, const ASN1_ITEM *it,
282  void *exarg)
283 {
284  ASN1_STREAM_ARG *sarg = exarg;
285  CMS_ContentInfo *cms = NULL;
286  if (pval)
287  cms = (CMS_ContentInfo *)*pval;
288  else
289  return 1;
290  switch (operation) {
291 
292  case ASN1_OP_STREAM_PRE:
293  if (CMS_stream(&sarg->boundary, cms) <= 0)
294  return 0;
295  case ASN1_OP_DETACHED_PRE:
296  sarg->ndef_bio = CMS_dataInit(cms, sarg->out);
297  if (!sarg->ndef_bio)
298  return 0;
299  break;
300 
301  case ASN1_OP_STREAM_POST:
302  case ASN1_OP_DETACHED_POST:
303  if (CMS_dataFinal(cms, sarg->ndef_bio) <= 0)
304  return 0;
305  break;
306 
307  }
308  return 1;
309 }

374 int CMS_SharedInfo_encode(unsigned char **pder, X509_ALGOR *kekalg,
375  ASN1_OCTET_STRING *ukm, int keylen)
376 {
377  union {
378  CMS_SharedInfo *pecsi;
379  ASN1_VALUE *a;
380  } intsi = {
381  NULL
382  };
383 
384  ASN1_OCTET_STRING oklen;
385  unsigned char kl[4];
386  CMS_SharedInfo ecsi;
387 
388  keylen <<= 3;
389  kl[0] = (keylen >> 24) & 0xff;
390  kl[1] = (keylen >> 16) & 0xff;
391  kl[2] = (keylen >> 8) & 0xff;
392  kl[3] = keylen & 0xff;
393  oklen.length = 4;
394  oklen.data = kl;
395  oklen.type = V_ASN1_OCTET_STRING;
396  oklen.flags = 0;
397  ecsi.keyInfo = kekalg;
398  ecsi.entityUInfo = ukm;
399  ecsi.suppPubInfo = &oklen;
400  intsi.pecsi = &ecsi;
401  return ASN1_item_i2d(intsi.a, pder, ASN1_ITEM_rptr(CMS_SharedInfo));
402 }
