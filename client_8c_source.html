60 int FuzzerTestOneInput(const uint8_t *buf, size_t len)
61 {
62  SSL *client;
63  BIO *in;
64  BIO *out;
65  SSL_CTX *ctx;
66 
67  if (len == 0)
68  return 0;
69 
70  /*
71  * TODO: use the ossltest engine (optionally?) to disable crypto checks.
72  */
73 
74  /* This only fuzzes the initial flow from the client so far. */
75  ctx = SSL_CTX_new(SSLv23_method());
76 
77  client = SSL_new(ctx);
78  OPENSSL_assert(SSL_set_cipher_list(client, "ALL:eNULL:@SECLEVEL=0") == 1);
79  SSL_set_tlsext_host_name(client, "localhost");
80  in = BIO_new(BIO_s_mem());
81  out = BIO_new(BIO_s_mem());
82  SSL_set_bio(client, in, out);
83  SSL_set_connect_state(client);
84  OPENSSL_assert((size_t)BIO_write(in, buf, len) == len);
85  if (SSL_do_handshake(client) == 1) {
86  /* Keep reading application data until error or EOF. */
87  uint8_t tmp[1024];
88  for (;;) {
89  if (SSL_read(client, tmp, sizeof(tmp)) <= 0) {
90  break;
91  }
92  }
93  }
94  SSL_free(client);
95  ERR_clear_error();
96  SSL_CTX_free(ctx);
97 
98  return 0;
99 }
