20 /*
21  * Following definition aliases fopen to fopen64 on above mentioned
22  * platforms. This makes it possible to open and sequentially access files
23  * larger than 2GB from 32-bit application. It does not allow to traverse
24  * them beyond 2GB with fseek/ftell, but on the other hand *no* 32-bit
25  * platform permits that, not with fseek/ftell. Not to mention that breaking
26  * 2GB limit for seeking would require surgery to *our* API. But sequential
27  * access suffices for practical cases when you can run into large files,
28  * such as fingerprinting, so we can let API alone. For reference, the list
29  * of 32-bit platforms which allow for sequential access of large files
30  * without extra "magic" comprise *BSD, Darwin, IRIX...
31  */
32 # ifndef _FILE_OFFSET_BITS
33 # define _FILE_OFFSET_BITS 64

68 BIO *BIO_new_file(const char *filename, const char *mode)
69 {
70  BIO *ret;
71  FILE *file = openssl_fopen(filename, mode);
72  int fp_flags = BIO_CLOSE;
73 
74  if (strchr(mode, 'b') == NULL)
75  fp_flags |= BIO_FP_TEXT;
76 
77  if (file == NULL) {
78  SYSerr(SYS_F_FOPEN, get_last_sys_error());
79  ERR_add_error_data(5, "fopen('", filename, "','", mode, "')");
80  if (errno == ENOENT
81 # ifdef ENXIO
82  || errno == ENXIO
83 # endif
84  )
85  BIOerr(BIO_F_BIO_NEW_FILE, BIO_R_NO_SUCH_FILE);
86  else
87  BIOerr(BIO_F_BIO_NEW_FILE, ERR_R_SYS_LIB);
88  return (NULL);
89  }
90  if ((ret = BIO_new(BIO_s_file())) == NULL) {
91  fclose(file);
92  return (NULL);
93  }
94 
95  BIO_clear_flags(ret, BIO_FLAGS_UPLINK); /* we did fopen -> we disengage
96  * UPLINK */
97  BIO_set_fp(ret, file, fp_flags);
98  return (ret);
99 }

128 static int file_free(BIO *a)
129 {
130  if (a == NULL)
131  return (0);
132  if (a->shutdown) {
133  if ((a->init) && (a->ptr != NULL)) {
134  if (a->flags & BIO_FLAGS_UPLINK)
135  UP_fclose(a->ptr);
136  else
137  fclose(a->ptr);
138  a->ptr = NULL;
139  a->flags = BIO_FLAGS_UPLINK;
140  }
141  a->init = 0;
142  }
143  return (1);
144 }

146 static int file_read(BIO *b, char *out, int outl)
147 {
148  int ret = 0;
149 
150  if (b->init && (out != NULL)) {
151  if (b->flags & BIO_FLAGS_UPLINK)
152  ret = UP_fread(out, 1, (int)outl, b->ptr);
153  else
154  ret = fread(out, 1, (int)outl, (FILE *)b->ptr);
155  if (ret == 0
156  && (b->flags & BIO_FLAGS_UPLINK) ? UP_ferror((FILE *)b->ptr) :
157  ferror((FILE *)b->ptr)) {
158  SYSerr(SYS_F_FREAD, get_last_sys_error());
159  BIOerr(BIO_F_FILE_READ, ERR_R_SYS_LIB);
160  ret = -1;
161  }
162  }
163  return (ret);
164 }

166 static int file_write(BIO *b, const char *in, int inl)
167 {
168  int ret = 0;
169 
170  if (b->init && (in != NULL)) {
171  if (b->flags & BIO_FLAGS_UPLINK)
172  ret = UP_fwrite(in, (int)inl, 1, b->ptr);
173  else
174  ret = fwrite(in, (int)inl, 1, (FILE *)b->ptr);
175  if (ret)
176  ret = inl;
177  /* ret=fwrite(in,1,(int)inl,(FILE *)b->ptr); */
178  /*
179  * according to Tim Hudson <tjh@cryptsoft.com>, the commented out
180  * version above can cause 'inl' write calls under some stupid stdio
181  * implementations (VMS)
182  */
183  }
184  return (ret);
185 }

187 static long file_ctrl(BIO *b, int cmd, long num, void *ptr)
188 {
189  long ret = 1;
190  FILE *fp = (FILE *)b->ptr;
191  FILE **fpp;
192  char p[4];
193 
194  switch (cmd) {
195  case BIO_C_FILE_SEEK:
196  case BIO_CTRL_RESET:
197  if (b->flags & BIO_FLAGS_UPLINK)
198  ret = (long)UP_fseek(b->ptr, num, 0);
199  else
200  ret = (long)fseek(fp, num, 0);
201  break;
202  case BIO_CTRL_EOF:
203  if (b->flags & BIO_FLAGS_UPLINK)
204  ret = (long)UP_feof(fp);
205  else
206  ret = (long)feof(fp);
207  break;
208  case BIO_C_FILE_TELL:
209  case BIO_CTRL_INFO:
210  if (b->flags & BIO_FLAGS_UPLINK)
211  ret = UP_ftell(b->ptr);
212  else
213  ret = ftell(fp);
214  break;
215  case BIO_C_SET_FILE_PTR:
216  file_free(b);
217  b->shutdown = (int)num & BIO_CLOSE;
218  b->ptr = ptr;
219  b->init = 1;
220 # if BIO_FLAGS_UPLINK!=0
221 # if defined(__MINGW32__) && defined(__MSVCRT__) && !defined(_IOB_ENTRIES)
222 # define _IOB_ENTRIES 20
223 # endif
224  /* Safety net to catch purely internal BIO_set_fp calls */
225 # if defined(_MSC_VER) && _MSC_VER>=1900
226  if (ptr == stdin || ptr == stdout || ptr == stderr)
227  BIO_clear_flags(b, BIO_FLAGS_UPLINK);
228 # elif defined(_IOB_ENTRIES)
229  if ((size_t)ptr >= (size_t)stdin &&
230  (size_t)ptr < (size_t)(stdin + _IOB_ENTRIES))
231  BIO_clear_flags(b, BIO_FLAGS_UPLINK);
232 # endif
233 # endif
234 # ifdef UP_fsetmod
235  if (b->flags & BIO_FLAGS_UPLINK)
236  UP_fsetmod(b->ptr, (char)((num & BIO_FP_TEXT) ? 't' : 'b'));
237  else
238 # endif
239  {
240 # if defined(OPENSSL_SYS_WINDOWS)
241  int fd = _fileno((FILE *)ptr);
242  if (num & BIO_FP_TEXT)
243  _setmode(fd, _O_TEXT);
244  else
245  _setmode(fd, _O_BINARY);
246 # elif defined(OPENSSL_SYS_MSDOS)
247  int fd = fileno((FILE *)ptr);
248  /* Set correct text/binary mode */
249  if (num & BIO_FP_TEXT)
250  _setmode(fd, _O_TEXT);
251  /* Dangerous to set stdin/stdout to raw (unless redirected) */
252  else {
253  if (fd == STDIN_FILENO || fd == STDOUT_FILENO) {
254  if (isatty(fd) <= 0)
255  _setmode(fd, _O_BINARY);
256  } else
257  _setmode(fd, _O_BINARY);
258  }
259 # elif defined(OPENSSL_SYS_WIN32_CYGWIN)
260  int fd = fileno((FILE *)ptr);
261  if (num & BIO_FP_TEXT)
262  setmode(fd, O_TEXT);
263  else
264  setmode(fd, O_BINARY);
265 # endif
266  }
267  break;
268  case BIO_C_SET_FILENAME:
269  file_free(b);
270  b->shutdown = (int)num & BIO_CLOSE;
271  if (num & BIO_FP_APPEND) {
272  if (num & BIO_FP_READ)
273  OPENSSL_strlcpy(p, "a+", sizeof p);
274  else
275  OPENSSL_strlcpy(p, "a", sizeof p);
276  } else if ((num & BIO_FP_READ) && (num & BIO_FP_WRITE))
277  OPENSSL_strlcpy(p, "r+", sizeof p);
278  else if (num & BIO_FP_WRITE)
279  OPENSSL_strlcpy(p, "w", sizeof p);
280  else if (num & BIO_FP_READ)
281  OPENSSL_strlcpy(p, "r", sizeof p);
282  else {
283  BIOerr(BIO_F_FILE_CTRL, BIO_R_BAD_FOPEN_MODE);
284  ret = 0;
285  break;
286  }
287 # if defined(OPENSSL_SYS_MSDOS) || defined(OPENSSL_SYS_WINDOWS) || defined(OPENSSL_SYS_WIN32_CYGWIN)
288  if (!(num & BIO_FP_TEXT))
289  strcat(p, "b");
290  else
291  strcat(p, "t");
292 # endif
293  fp = openssl_fopen(ptr, p);
294  if (fp == NULL) {
295  SYSerr(SYS_F_FOPEN, get_last_sys_error());
296  ERR_add_error_data(5, "fopen('", ptr, "','", p, "')");
297  BIOerr(BIO_F_FILE_CTRL, ERR_R_SYS_LIB);
298  ret = 0;
299  break;
300  }
301  b->ptr = fp;
302  b->init = 1;
303  BIO_clear_flags(b, BIO_FLAGS_UPLINK); /* we did fopen -> we disengage
304  * UPLINK */
305  break;
306  case BIO_C_GET_FILE_PTR:
307  /* the ptr parameter is actually a FILE ** in this case. */
308  if (ptr != NULL) {
309  fpp = (FILE **)ptr;
310  *fpp = (FILE *)b->ptr;
311  }
312  break;
313  case BIO_CTRL_GET_CLOSE:
314  ret = (long)b->shutdown;
315  break;
316  case BIO_CTRL_SET_CLOSE:
317  b->shutdown = (int)num;
318  break;
319  case BIO_CTRL_FLUSH:
320  if (b->flags & BIO_FLAGS_UPLINK)
321  UP_fflush(b->ptr);
322  else
323  fflush((FILE *)b->ptr);
324  break;
325  case BIO_CTRL_DUP:
326  ret = 1;
327  break;
328 
329  case BIO_CTRL_WPENDING:
330  case BIO_CTRL_PENDING:
331  case BIO_CTRL_PUSH:
332  case BIO_CTRL_POP:
333  default:
334  ret = 0;
335  break;
336  }
337  return (ret);
338 }

340 static int file_gets(BIO *bp, char *buf, int size)
341 {
342  int ret = 0;
343 
344  buf[0] = '\0';
345  if (bp->flags & BIO_FLAGS_UPLINK) {
346  if (!UP_fgets(buf, size, bp->ptr))
347  goto err;
348  } else {
349  if (!fgets(buf, size, (FILE *)bp->ptr))
350  goto err;
351  }
352  if (buf[0] != '\0')
353  ret = strlen(buf);
354  err:
355  return (ret);
356 }
