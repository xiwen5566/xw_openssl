94 /* Preconditions:
95  * |h| bounded by 1.1*2^26,1.1*2^25,1.1*2^26,1.1*2^25,etc.
96  *
97  * Write p=2^255-19; q=floor(h/p).
98  * Basic claim: q = floor(2^(-255)(h + 19 2^(-25)h9 + 2^(-1))).
99  *
100  * Proof:
101  * Have |h|<=p so |q|<=1 so |19^2 2^(-255) q|<1/4.
102  * Also have |h-2^230 h9|<2^231 so |19 2^(-255)(h-2^230 h9)|<1/4.
103  *
104  * Write y=2^(-1)-19^2 2^(-255)q-19 2^(-255)(h-2^230 h9).
105  * Then 0<y<1.
106  *
107  * Write r=h-pq.
108  * Have 0<=r<=p-1=2^255-20.
109  * Thus 0<=r+19(2^-255)r<r+19(2^-255)2^255<=2^255-1.
110  *
111  * Write x=r+19(2^-255)r+y.
112  * Then 0<x<2^255 so floor(2^(-255)x) = 0 so floor(q+2^(-255)x) = q.
113  *
114  * Have q+2^(-255)x = 2^(-255)(h + 19 2^(-25) h9 + 2^(-1))
115  * so floor(2^(-255)(h + 19 2^(-25) h9 + 2^(-1))) = q. */

125  int32_t h8 = h[8];
126  int32_t h9 = h[9];
127  int32_t q;
128 
129  q = (19 * h9 + (((int32_t) 1) << 24)) >> 25;
130  q = (h0 + q) >> 26;
131  q = (h1 + q) >> 25;
132  q = (h2 + q) >> 26;
133  q = (h3 + q) >> 25;
134  q = (h4 + q) >> 26;
135  q = (h5 + q) >> 25;
136  q = (h6 + q) >> 26;
137  q = (h7 + q) >> 25;
138  q = (h8 + q) >> 26;
139  q = (h9 + q) >> 25;
140 
141  /* Goal: Output h-(2^255-19)q, which is between 0 and 2^255-20. */
142  h0 += 19 * q;
143  /* Goal: Output h-2^255 q, which is between 0 and 2^255-20. */
144 
145  h1 += h0 >> 26; h0 &= kBottom26Bits;
146  h2 += h1 >> 25; h1 &= kBottom25Bits;
147  h3 += h2 >> 26; h2 &= kBottom26Bits;
148  h4 += h3 >> 25; h3 &= kBottom25Bits;
149  h5 += h4 >> 26; h4 &= kBottom26Bits;
150  h6 += h5 >> 25; h5 &= kBottom25Bits;
151  h7 += h6 >> 26; h6 &= kBottom26Bits;
152  h8 += h7 >> 25; h7 &= kBottom25Bits;
153  h9 += h8 >> 26; h8 &= kBottom26Bits;
154  h9 &= kBottom25Bits;
155  /* h10 = carry9 */
156 
157  /* Goal: Output h0+...+2^255 h10-2^255 q, which is between 0 and 2^255-20.
158  * Have h0+...+2^230 h9 between 0 and 2^255-1;
159  * evidently 2^255 h10-2^255 q = 0.
160  * Goal: Output h0+...+2^230 h9. */
161 
162  s[0] = h0 >> 0;
163  s[1] = h0 >> 8;
164  s[2] = h0 >> 16;
165  s[3] = (h0 >> 24) | ((uint32_t)(h1) << 2);
166  s[4] = h1 >> 6;
167  s[5] = h1 >> 14;
168  s[6] = (h1 >> 22) | ((uint32_t)(h2) << 3);
169  s[7] = h2 >> 5;
170  s[8] = h2 >> 13;
171  s[9] = (h2 >> 21) | ((uint32_t)(h3) << 5);
172  s[10] = h3 >> 3;
173  s[11] = h3 >> 11;
174  s[12] = (h3 >> 19) | ((uint32_t)(h4) << 6);
175  s[13] = h4 >> 2;
176  s[14] = h4 >> 10;
177  s[15] = h4 >> 18;
178  s[16] = h5 >> 0;
179  s[17] = h5 >> 8;
180  s[18] = h5 >> 16;
181  s[19] = (h5 >> 24) | ((uint32_t)(h6) << 1);
182  s[20] = h6 >> 7;
183  s[21] = h6 >> 15;
184  s[22] = (h6 >> 23) | ((uint32_t)(h7) << 3);
185  s[23] = h7 >> 5;
186  s[24] = h7 >> 13;
187  s[25] = (h7 >> 21) | ((uint32_t)(h8) << 4);
188  s[26] = h8 >> 4;
189  s[27] = h8 >> 12;
190  s[28] = (h8 >> 20) | ((uint32_t)(h9) << 6);
191  s[29] = h9 >> 2;
192  s[30] = h9 >> 10;
193  s[31] = h9 >> 18;
194 }

242 /* h = f * g
243  * Can overlap h with f or g.
244  *
245  * Preconditions:
246  * |f| bounded by 1.65*2^26,1.65*2^25,1.65*2^26,1.65*2^25,etc.
247  * |g| bounded by 1.65*2^26,1.65*2^25,1.65*2^26,1.65*2^25,etc.

249  * Postconditions:
250  * |h| bounded by 1.01*2^25,1.01*2^24,1.01*2^25,1.01*2^24,etc.
251  *
252  * Notes on implementation strategy:
253  *
254  * Using schoolbook multiplication.
255  * Karatsuba would save a little in some cost models.
256  *
257  * Most multiplications by 2 and 19 are 32-bit precomputations;
258  * cheaper than 64-bit postcomputations.
259  *
260  * There is one remaining multiplication by 19 in the carry chain;
261  * one *19 precomputation can be merged into this,
262  * but the resulting data flow is considerably less clean.
263  *
264  * There are 12 carries below.
265  * 10 of them are 2-way parallelizable and vectorizable.
266  * Can get away with 11 carries, but then data flow is much deeper.
267  *
268  * With tighter constraints on inputs can squeeze carries into int32. */
