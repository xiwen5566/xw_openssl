142 /* Give a hash of the currently set handshake */
143 int ssl_handshake_hash(SSL *s, unsigned char *out, size_t outlen,
144  size_t *hashlen)
145 {
146  if (sizeof(hs_start_hash) > outlen
147  || sizeof(hs_full_hash) != sizeof(hs_start_hash))
148  return 0;
149 
150  if (full_hash) {
151  memcpy(out, hs_full_hash, sizeof(hs_full_hash));
152  *hashlen = sizeof(hs_full_hash);
153  } else {
154  memcpy(out, hs_start_hash, sizeof(hs_start_hash));
155  *hashlen = sizeof(hs_start_hash);
156  }
157 
158  return 1;
159 }

197 static int test_secret(SSL *s, unsigned char *prk,
198  const unsigned char *label, size_t labellen,
199  const unsigned char *ref_secret,
200  const unsigned char *ref_key, const unsigned char *ref_iv)
201 {
202  size_t hashsize;
203  unsigned char gensecret[EVP_MAX_MD_SIZE];
204  unsigned char hash[EVP_MAX_MD_SIZE];
205  unsigned char key[KEYLEN];
206  unsigned char iv[IVLEN];
207  const EVP_MD *md = ssl_handshake_md(s);
208 
209  if (!ssl_handshake_hash(s, hash, sizeof(hash), &hashsize)) {
210  fprintf(stderr, "Failed to get hash\n");
211  return 0;
212  }
213 
214  if (!tls13_hkdf_expand(s, md, prk, label, labellen, hash, gensecret,
215  hashsize)) {
216  fprintf(stderr, "Secret generation failed\n");
217  return 0;
218  }
219 
220  if (memcmp(gensecret, ref_secret, hashsize) != 0) {
221  fprintf(stderr, "Generated secret does not match\n");
222  return 0;
223  }
224 
225  if (!tls13_derive_key(s, gensecret, key, KEYLEN)) {
226  fprintf(stderr, "Key generation failed\n");
227  return 0;
228  }
229 
230  if (memcmp(key, ref_key, KEYLEN) != 0) {
231  fprintf(stderr, "Generated key does not match\n");
232  return 0;
233  }
234 
235  if (!tls13_derive_iv(s, gensecret, iv, IVLEN)) {
236  fprintf(stderr, "IV generation failed\n");
237  return 0;
238  }
239 
240  if (memcmp(iv, ref_iv, IVLEN) != 0) {
241  fprintf(stderr, "Generated IV does not match\n");
242  return 0;
243  }
244 
245  return 1;
246 }

248 static int test_handshake_secrets(void)
249 {
250  SSL_CTX *ctx = NULL;
251  SSL *s = NULL;
252  int ret = 0;
253  size_t hashsize;
254  unsigned char out_master_secret[EVP_MAX_MD_SIZE];
255  size_t master_secret_length;
256 
257  ctx = SSL_CTX_new(TLS_method());
258  if (ctx == NULL)
259  goto err;
260 
261  s = SSL_new(ctx);
262  if (s == NULL)
263  goto err;
264 
265  s->session = SSL_SESSION_new();
266  if (s->session == NULL)
267  goto err;
268 
269  if (!tls13_generate_secret(s, ssl_handshake_md(s), NULL, NULL, 0,
270  (unsigned char *)&s->early_secret)) {
271  fprintf(stderr, "Early secret generation failed\n");
272  goto err;
273  }
274 
275  if (memcmp(s->early_secret, early_secret, sizeof(early_secret)) != 0) {
276  fprintf(stderr, "Early secret does not match\n");
277  goto err;
278  }
279 
280  if (!tls13_generate_handshake_secret(s, ecdhe_secret,
281  sizeof(ecdhe_secret))) {
282  fprintf(stderr, "Hanshake secret generation failed\n");
283  goto err;
284  }
285 
286  if (memcmp(s->handshake_secret, handshake_secret,
287  sizeof(handshake_secret)) != 0) {
288  fprintf(stderr, "Handshake secret does not match\n");
289  goto err;
290  }
291 
292  hashsize = EVP_MD_size(ssl_handshake_md(s));
293  if (sizeof(client_hts) != hashsize || sizeof(client_hts_key) != KEYLEN
294  || sizeof(client_hts_iv) != IVLEN) {
295  fprintf(stderr, "Internal test error\n");
296  goto err;
297  }
298 
299  if (!test_secret(s, s->handshake_secret, (unsigned char *)client_hts_label,
300  strlen(client_hts_label), client_hts, client_hts_key,
301  client_hts_iv)) {
302  fprintf(stderr, "Client handshake secret test failed\n");
303  goto err;
304  }
305 
306  if (sizeof(server_hts) != hashsize || sizeof(server_hts_key) != KEYLEN
307  || sizeof(server_hts_iv) != IVLEN) {
308  fprintf(stderr, "Internal test error\n");
309  goto err;
310  }
311 
312  if (!test_secret(s, s->handshake_secret, (unsigned char *)server_hts_label,
313  strlen(server_hts_label), server_hts, server_hts_key,
314  server_hts_iv)) {
315  fprintf(stderr, "Server handshake secret test failed\n");
316  goto err;
317  }
318 
319  /*
320  * Ensure the mocked out ssl_handshake_hash() returns the full handshake
321  * hash.
322  */
323  full_hash = 1;
324 
325  if (!tls13_generate_master_secret(s, out_master_secret,
326  s->handshake_secret, hashsize,
327  &master_secret_length)) {
328  fprintf(stderr, "Master secret generation failed\n");
329  goto err;
330  }
331 
332  if (master_secret_length != sizeof(master_secret) ||
333  memcmp(out_master_secret, master_secret,
334  sizeof(master_secret)) != 0) {
335  fprintf(stderr, "Master secret does not match\n");
336  goto err;
337  }
338 
339  if (sizeof(client_ats) != hashsize || sizeof(client_ats_key) != KEYLEN
340  || sizeof(client_ats_iv) != IVLEN) {
341  fprintf(stderr, "Internal test error\n");
342  goto err;
343  }
344 
345  if (!test_secret(s, out_master_secret, (unsigned char *)client_ats_label,
346  strlen(client_ats_label), client_ats, client_ats_key,
347  client_ats_iv)) {
348  fprintf(stderr, "Client application data secret test failed\n");
349  goto err;
350  }
351 
352  if (sizeof(server_ats) != hashsize || sizeof(server_ats_key) != KEYLEN
353  || sizeof(server_ats_iv) != IVLEN) {
354  fprintf(stderr, "Internal test error\n");
355  goto err;
356  }
357 
358  if (!test_secret(s, out_master_secret, (unsigned char *)server_ats_label,
359  strlen(server_ats_label), server_ats, server_ats_key,
360  server_ats_iv)) {
361  fprintf(stderr, "Server application data secret test failed\n");
362  goto err;
363  }
364 
365  ret = 1;
366  err:
367  SSL_free(s);
368  SSL_CTX_free(ctx);
369  return ret;
370 }
