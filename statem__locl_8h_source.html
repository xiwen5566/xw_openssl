56 /* Message processing return codes */
57 typedef enum {
58  /* Something bad happened */
59  MSG_PROCESS_ERROR,
60  /* We've finished reading - swap to writing */
61  MSG_PROCESS_FINISHED_READING,
62  /*
63  * We've completed the main processing of this message but there is some
64  * post processing to be done.
65  */
66  MSG_PROCESS_CONTINUE_PROCESSING,
67  /* We've finished this message - read the next message */
68  MSG_PROCESS_CONTINUE_READING
69 } MSG_PROCESS_RETURN;
