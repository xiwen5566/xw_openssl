34  "string to make the random number generator think it has entropy";

36 static int run_rfc5114_tests(void);
37 
38 int main(int argc, char *argv[])
39 {
40  BN_GENCB *_cb = NULL;
41  DH *a = NULL;
42  DH *b = NULL;
43  const BIGNUM *ap = NULL, *ag = NULL, *apub_key = NULL, *priv_key = NULL;
44  const BIGNUM *bpub_key = NULL;
45  BIGNUM *bp = NULL, *bg = NULL;
46  char buf[12] = {0};
47  unsigned char *abuf = NULL;
48  unsigned char *bbuf = NULL;
49  int i, alen, blen, aout, bout;
50  int ret = 1;
51  BIO *out = NULL;
52 
53  CRYPTO_set_mem_debug(1);
54  CRYPTO_mem_ctrl(CRYPTO_MEM_CHECK_ON);
55 
56  RAND_seed(rnd_seed, sizeof rnd_seed);
57 
58  out = BIO_new(BIO_s_file());
59  if (out == NULL)
60  EXIT(1);
61  BIO_set_fp(out, stdout, BIO_NOCLOSE | BIO_FP_TEXT);
62 
63  _cb = BN_GENCB_new();
64  if (_cb == NULL)
65  goto err;
66  BN_GENCB_set(_cb, &cb, out);
67  if (((a = DH_new()) == NULL)
68  || (!DH_generate_parameters_ex(a, 64, DH_GENERATOR_5, _cb)))
69  goto err;
70 
71  if (!DH_check(a, &i))
72  goto err;
73  if (i & DH_CHECK_P_NOT_PRIME)
74  BIO_puts(out, "p value is not prime\n");
75  if (i & DH_CHECK_P_NOT_SAFE_PRIME)
76  BIO_puts(out, "p value is not a safe prime\n");
77  if (i & DH_UNABLE_TO_CHECK_GENERATOR)
78  BIO_puts(out, "unable to check the generator value\n");
79  if (i & DH_NOT_SUITABLE_GENERATOR)
80  BIO_puts(out, "the g value is not a generator\n");
81 
82  DH_get0_pqg(a, &ap, NULL, &ag);
83  BIO_puts(out, "\np =");
84  BN_print(out, ap);
85  BIO_puts(out, "\ng =");
86  BN_print(out, ag);
87  BIO_puts(out, "\n");
88 
89  b = DH_new();
90  if (b == NULL)
91  goto err;
92 
93  bp = BN_dup(ap);
94  bg = BN_dup(ag);
95  if ((bp == NULL) || (bg == NULL) || !DH_set0_pqg(b, bp, NULL, bg))
96  goto err;
97  bp = bg = NULL;
98 
99  if (!DH_generate_key(a))
100  goto err;
101  DH_get0_key(a, &apub_key, &priv_key);
102  BIO_puts(out, "pri 1=");
103  BN_print(out, priv_key);
104  BIO_puts(out, "\npub 1=");
105  BN_print(out, apub_key);
106  BIO_puts(out, "\n");
107 
108  if (!DH_generate_key(b))
109  goto err;
110  DH_get0_key(b, &bpub_key, &priv_key);
111  BIO_puts(out, "pri 2=");
112  BN_print(out, priv_key);
113  BIO_puts(out, "\npub 2=");
114  BN_print(out, bpub_key);
115  BIO_puts(out, "\n");
116 
117  alen = DH_size(a);
118  abuf = OPENSSL_malloc(alen);
119  if (abuf == NULL)
120  goto err;
121 
122  aout = DH_compute_key(abuf, bpub_key, a);
123 
124  BIO_puts(out, "key1 =");
125  for (i = 0; i < aout; i++) {
126  sprintf(buf, "%02X", abuf[i]);
127  BIO_puts(out, buf);
128  }
129  BIO_puts(out, "\n");
130 
131  blen = DH_size(b);
132  bbuf = OPENSSL_malloc(blen);
133  if (bbuf == NULL)
134  goto err;
135 
136  bout = DH_compute_key(bbuf, apub_key, b);
137 
138  BIO_puts(out, "key2 =");
139  for (i = 0; i < bout; i++) {
140  sprintf(buf, "%02X", bbuf[i]);
141  BIO_puts(out, buf);
142  }
143  BIO_puts(out, "\n");
144  if ((aout < 4) || (bout != aout) || (memcmp(abuf, bbuf, aout) != 0)) {
145  fprintf(stderr, "Error in DH routines\n");
146  ret = 1;
147  } else
148  ret = 0;
149  if (!run_rfc5114_tests())
150  ret = 1;
151  err:
152  (void)BIO_flush(out);
153  ERR_print_errors_fp(stderr);
154 
155  OPENSSL_free(abuf);
156  OPENSSL_free(bbuf);
157  DH_free(b);
158  DH_free(a);
159  BN_free(bp);
160  BN_free(bg);
161  BN_GENCB_free(_cb);
162  BIO_free(out);
163 
164 #ifndef OPENSSL_NO_CRYPTO_MDEBUG
165  if (CRYPTO_mem_leaks_fp(stderr) <= 0)
166  ret = 1;
167 #endif
168 
169  EXIT(ret);
170 }

172 static int cb(int p, int n, BN_GENCB *arg)
173 {
174  char c = '*';
175 
176  if (p == 0)
177  c = '.';
178  if (p == 1)
179  c = '+';
180  if (p == 2)
181  c = '*';
182  if (p == 3)
183  c = '\n';
184  BIO_write(BN_GENCB_get_arg(arg), &c, 1);
185  (void)BIO_flush(BN_GENCB_get_arg(arg));
186  return 1;
187 }

471 static int run_rfc5114_tests(void)
472 {
473  int i;
474  DH *dhA = NULL;
475  DH *dhB = NULL;
476  unsigned char *Z1 = NULL;
477  unsigned char *Z2 = NULL;
478  const rfc5114_td *td = NULL;
479  BIGNUM *bady = NULL, *priv_key = NULL, *pub_key = NULL;
480  const BIGNUM *pub_key_tmp;
481 
482  for (i = 0; i < (int)OSSL_NELEM(rfctd); i++) {
483  td = rfctd + i;
484  /* Set up DH structures setting key components */
485  dhA = td->get_param();
486  dhB = td->get_param();
487  if ((dhA == NULL) || (dhB == NULL))
488  goto bad_err;
489 
490  priv_key = BN_bin2bn(td->xA, td->xA_len, NULL);
491  pub_key = BN_bin2bn(td->yA, td->yA_len, NULL);
492  if (priv_key == NULL || pub_key == NULL
493  || !DH_set0_key(dhA, pub_key, priv_key))
494  goto bad_err;
495 
496  priv_key = BN_bin2bn(td->xB, td->xB_len, NULL);
497  pub_key = BN_bin2bn(td->yB, td->yB_len, NULL);
498 
499  if (priv_key == NULL || pub_key == NULL
500  || !DH_set0_key(dhB, pub_key, priv_key))
501  goto bad_err;
502  priv_key = pub_key = NULL;
503 
504  if ((td->Z_len != (size_t)DH_size(dhA))
505  || (td->Z_len != (size_t)DH_size(dhB)))
506  goto err;
507 
508  Z1 = OPENSSL_malloc(DH_size(dhA));
509  Z2 = OPENSSL_malloc(DH_size(dhB));
510  if ((Z1 == NULL) || (Z2 == NULL))
511  goto bad_err;
512  /*
513  * Work out shared secrets using both sides and compare with expected
514  * values.
515  */
516  DH_get0_key(dhB, &pub_key_tmp, NULL);
517  if (DH_compute_key(Z1, pub_key_tmp, dhA) == -1)
518  goto bad_err;
519 
520  DH_get0_key(dhA, &pub_key_tmp, NULL);
521  if (DH_compute_key(Z2, pub_key_tmp, dhB) == -1)
522  goto bad_err;
523 
524  if (memcmp(Z1, td->Z, td->Z_len))
525  goto err;
526  if (memcmp(Z2, td->Z, td->Z_len))
527  goto err;
528 
529  printf("RFC5114 parameter test %d OK\n", i + 1);
530 
531  DH_free(dhA);
532  DH_free(dhB);
533  OPENSSL_free(Z1);
534  OPENSSL_free(Z2);
535  dhA = NULL;
536  dhB = NULL;
537  Z1 = NULL;
538  Z2 = NULL;
539  }
540 
541  /* Now i == OSSL_NELEM(rfctd) */
542  /* RFC5114 uses unsafe primes, so now test an invalid y value */
543  dhA = DH_get_2048_224();
544  if (dhA == NULL)
545  goto bad_err;
546  Z1 = OPENSSL_malloc(DH_size(dhA));
547  if (Z1 == NULL)
548  goto bad_err;
549 
550  bady = BN_bin2bn(dhtest_rfc5114_2048_224_bad_y,
551  sizeof(dhtest_rfc5114_2048_224_bad_y), NULL);
552  if (bady == NULL)
553  goto bad_err;
554 
555  if (!DH_generate_key(dhA))
556  goto bad_err;
557 
558  if (DH_compute_key(Z1, bady, dhA) != -1) {
559  /*
560  * DH_compute_key should fail with -1. If we get here we unexpectedly
561  * allowed an invalid y value
562  */
563  goto err;
564  }
565  /* We'll have a stale error on the queue from the above test so clear it */
566  ERR_clear_error();
567 
568  printf("RFC5114 parameter test %d OK\n", i + 1);
569 
570  BN_free(bady);
571  DH_free(dhA);
572  OPENSSL_free(Z1);
573 
574  return 1;
575  bad_err:
576  BN_free(bady);
577  DH_free(dhA);
578  DH_free(dhB);
579  BN_free(pub_key);
580  BN_free(priv_key);
581  OPENSSL_free(Z1);
582  OPENSSL_free(Z2);
583 
584  fprintf(stderr, "Initialisation error RFC5114 set %d\n", i + 1);
585  ERR_print_errors_fp(stderr);
586  return 0;
587  err:
588  BN_free(bady);
589  DH_free(dhA);
590  DH_free(dhB);
591  OPENSSL_free(Z1);
592  OPENSSL_free(Z2);
593 
594  fprintf(stderr, "Test failed RFC5114 set %d\n", i + 1);
595  return 0;
596 }
