92 {

95  DES_LONG out[2], ll;

97  DES_key_schedule ks;
98  unsigned char bb[9];

101 
102  x = ret[0] = salt[0];
103  if (x == 0 || x >= sizeof(con_salt))
104  return NULL;

106  x = ret[1] = salt[1];
107  if (x == 0 || x >= sizeof(con_salt))
108  return NULL;

110 
111  /*
112  * EAY r=strlen(buf); r=(r+7)/8;
113  */
114  for (i = 0; i < 8; i++) {
115  c = *(buf++);
116  if (!c)
117  break;
118  key[i] = (c << 1);
119  }
120  for (; i < 8; i++)
121  key[i] = 0;
122 
123  DES_set_key_unchecked(&key, &ks);
124  fcrypt_body(&(out[0]), &ks, Eswap0, Eswap1);
125

128  ll = out[1];
129  l2c(ll, b);
130  y = 0;
131  u = 0x80;

134  c = 0;
