133 static int ts_resp_set_tst_info(TS_RESP *a)
134 {
135  long status;
136 
137  status = ASN1_INTEGER_get(a->status_info->status);
138 
139  if (a->token) {
140  if (status != 0 && status != 1) {
141  TSerr(TS_F_TS_RESP_SET_TST_INFO, TS_R_TOKEN_PRESENT);
142  return 0;
143  }
144  TS_TST_INFO_free(a->tst_info);
145  a->tst_info = PKCS7_to_TS_TST_INFO(a->token);
146  if (!a->tst_info) {
147  TSerr(TS_F_TS_RESP_SET_TST_INFO,
148  TS_R_PKCS7_TO_TS_TST_INFO_FAILED);
149  return 0;
150  }
151  } else if (status == 0 || status == 1) {
152  TSerr(TS_F_TS_RESP_SET_TST_INFO, TS_R_TOKEN_NOT_PRESENT);
153  return 0;
154  }
155 
156  return 1;
157 }

159 static int ts_resp_cb(int op, ASN1_VALUE **pval, const ASN1_ITEM *it,
160  void *exarg)
161 {
162  TS_RESP *ts_resp = (TS_RESP *)*pval;
163  if (op == ASN1_OP_NEW_POST) {
164  ts_resp->tst_info = NULL;
165  } else if (op == ASN1_OP_FREE_POST) {
166  TS_TST_INFO_free(ts_resp->tst_info);
167  } else if (op == ASN1_OP_D2I_POST) {
168  if (ts_resp_set_tst_info(ts_resp) == 0)
169  return 0;
170  }
171  return 1;
172 }

228 /* Getting encapsulated TS_TST_INFO object from PKCS7. */
229 TS_TST_INFO *PKCS7_to_TS_TST_INFO(PKCS7 *token)
230 {
231  PKCS7_SIGNED *pkcs7_signed;
232  PKCS7 *enveloped;
233  ASN1_TYPE *tst_info_wrapper;
234  ASN1_OCTET_STRING *tst_info_der;
235  const unsigned char *p;
236 
237  if (!PKCS7_type_is_signed(token)) {
238  TSerr(TS_F_PKCS7_TO_TS_TST_INFO, TS_R_BAD_PKCS7_TYPE);
239  return NULL;
240  }
241  if (PKCS7_get_detached(token)) {
242  TSerr(TS_F_PKCS7_TO_TS_TST_INFO, TS_R_DETACHED_CONTENT);
243  return NULL;
244  }
245  pkcs7_signed = token->d.sign;
246  enveloped = pkcs7_signed->contents;
247  if (OBJ_obj2nid(enveloped->type) != NID_id_smime_ct_TSTInfo) {
248  TSerr(TS_F_PKCS7_TO_TS_TST_INFO, TS_R_BAD_PKCS7_TYPE);
249  return NULL;
250  }
251  tst_info_wrapper = enveloped->d.other;
252  if (tst_info_wrapper->type != V_ASN1_OCTET_STRING) {
253  TSerr(TS_F_PKCS7_TO_TS_TST_INFO, TS_R_BAD_TYPE);
254  return NULL;
255  }
256  tst_info_der = tst_info_wrapper->value.octet_string;
257  p = tst_info_der->data;
258  return d2i_TS_TST_INFO(NULL, &p, tst_info_der->length);
259 }
