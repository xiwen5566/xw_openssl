96 static AUTHORITY_INFO_ACCESS *v2i_AUTHORITY_INFO_ACCESS(X509V3_EXT_METHOD

121  if (!ptmp) {

126  objlen = ptmp - cnf->name;
127  ctmp.name = ptmp + 1;
128  ctmp.value = cnf->value;

144  OPENSSL_free(objtmp);

146  }
