20 static ossl_inline uint32_t load32(const uint8_t *src)
21 {
22  const union {
23  long one;
24  char little;
25  } is_endian = { 1 };
26 
27  if (is_endian.little) {
28  uint32_t w;
29  memcpy(&w, src, sizeof(w));
30  return w;
31  } else {
32  uint32_t w = ((uint32_t)src[0])
33  | ((uint32_t)src[1] << 8)
34  | ((uint32_t)src[2] << 16)
35  | ((uint32_t)src[3] << 24);
36  return w;
37  }
38 }
