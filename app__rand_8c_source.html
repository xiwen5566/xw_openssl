67  for (p = name; ((*p != '\0') && (*p != LIST_SEPARATOR_CHAR)); p++) ;
68  if (*p == '\0')
69  last = 1;
70  *p = '\0';

73  if (*n == '\0')
74  break;
75 
76 #ifndef OPENSSL_NO_EGD

78  if (egd > 0)

80  else
81 #endif

85  }
