42 /*
43  * It has come to my attention that there are 2 versions of the RC2 key
44  * schedule. One which is normal, and anther which has a hook to use a
45  * reduced key length. BSAFE uses the 'retarded' version. What I previously
46  * shipped is the same as specifying 1024 for the 'bits' parameter. Bsafe
47  * uses a version where the bits parameter is the same as len*8
48  */
49 void RC2_set_key(RC2_KEY *key, int len, const unsigned char *data, int bits)
50 {
51  int i, j;
52  unsigned char *k;
53  RC2_INT *ki;
54  unsigned int c, d;
55 
56  k = (unsigned char *)&(key->data[0]);
57  *k = 0; /* for if there is a zero length key */
58 
59  if (len > 128)
60  len = 128;
61  if (bits <= 0)
62  bits = 1024;
63  if (bits > 1024)
64  bits = 1024;
65 
66  for (i = 0; i < len; i++)
67  k[i] = data[i];
68 
69  /* expand table */
70  d = k[len - 1];
71  j = 0;
72  for (i = len; i < 128; i++, j++) {
73  d = key_table[(k[j] + d) & 0xff];
74  k[i] = d;
75  }
76 
77  /* hmm.... key reduction to 'bits' bits */
78 
79  j = (bits + 7) >> 3;
80  i = 128 - j;
81  c = (0xff >> (-bits & 0x07));
82 
83  d = key_table[k[i] & c];
84  k[i] = d;
85  while (i--) {
86  d = key_table[k[i + j] ^ d];
87  k[i] = d;
88  }
89 
90  /* copy from bytes into RC2_INT's */
91  ki = &(key->data[63]);
92  for (i = 127; i >= 0; i -= 2)
93  *(ki--) = ((k[i] << 8) | k[i - 1]) & 0xffff;
94 }
