76 template <typename T>
77 struct Deleter {
78  void operator()(T *ptr) {
79  // Rather than specialize Deleter for each type, we specialize
80  // DeleterImpl. This allows bssl::UniquePtr<T> to be used while only
81  // including base.h as long as the destructor is not emitted. This matches
82  // std::unique_ptr's behavior on forward-declared types.
83  //
84  // DeleterImpl itself is specialized in the corresponding module's header
85  // and must be included to release an object. If not included, the compiler
86  // will error that DeleterImpl<T> does not have a method Free.
87  DeleterImpl<T>::Free(ptr);
88  }
89 };

91 template <typename T, typename CleanupRet, void (*init)(T *),
92  CleanupRet (*cleanup)(T *)>
93 class StackAllocated {
94  public:
95  StackAllocated() { init(&ctx_); }

99  T& operator=(const StackAllocated<T, CleanupRet, init, cleanup> &) = delete;
100 
101  T *get() { return &ctx_; }
102  const T *get() const { return &ctx_; }

123 // This makes a unique_ptr to STACK_OF(type) that owns all elements on the
124 // stack, i.e. it uses sk_pop_free() to clean up.
125 #define BORINGSSL_MAKE_STACK_DELETER(type, deleter) \
126  namespace internal { \
127  template <> \
128  struct DeleterImpl<STACK_OF(type)> { \
129  static void Free(STACK_OF(type) *ptr) { \
130  sk_##type##_pop_free(ptr, deleter); \
131  } \
132  }; \
133  }
